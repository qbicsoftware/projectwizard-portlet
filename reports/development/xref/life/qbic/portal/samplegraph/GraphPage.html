30    private Map<String, ISampleBean> currentSampleMap;
31    private ComboBox factorBox;
32  
33    private ProjectGraph sampleGraph;
34  
35    public GraphPage() {
36      setSpacing(true);
37    }

39    public void setProjectGraph(StructuredExperiment graph, Map<String, ISampleBean> idsToSamples) {
40      this.currentSampleMap = idsToSamples;
41      this.factorBox = new ComboBox("Experimental Factor");
42      factorBox.setVisible(false);
43      addComponent(factorBox);
44  
45      if (!graph.getFactorsToSamples().isEmpty()) {
46        factorBox.addItems(graph.getFactorsToSamples().keySet());
47        factorBox.setVisible(true);
48      }
49  
50      factorBox.setImmediate(true);
51      factorBox.setNullSelectionAllowed(false);
52      final GraphPage parent = this;
53      factorBox.addValueChangeListener(new ValueChangeListener() {
54  
55        @Override
56        public void valueChange(ValueChangeEvent event) {
57          Object factor = factorBox.getValue();
58          if (sampleGraph != null)
59            parent.removeComponent(sampleGraph);
60          sampleGraph = new ProjectGraph(parent, ProjectWizardUI.getPathToVaadinFolder()+"img/");
61          sampleGraph.setSizeFull();
62          parent.addComponent(sampleGraph);
63          if (factor != null) {
64            sampleGraph.setProject(graph.getFactorsToSamples().get(factor));
65          }
66        }
67      });
68    }
69  
70    public void showDatasetsForSamples(String label, List<String> sampleCodes) {
71      Window subWindow = new Window(" " + label + " information");
72      subWindow.setWidth("680px");
73  
74      VerticalLayout layout = new VerticalLayout();
75      layout.setSpacing(true);
76      layout.setMargin(true);
77      List<ISampleBean> samples = new ArrayList<ISampleBean>();
78  
79      for (String code : sampleCodes) {
80        ISampleBean s = currentSampleMap.get(code);
81        samples.add(s);
82      }
83  
84      Table haveData = new Table("Samples/Entities");
85      haveData.setStyleName(ValoTheme.TABLE_SMALL);
86      haveData.addContainerProperty("Sample", String.class, null);
87      haveData.addContainerProperty("Secondary Name", String.class, null);
88      haveData.addContainerProperty("Lab ID", String.class, null);
89      int i = 0;
90      for (ISampleBean s : samples) {
91        i++;
92        List<Object> row = new ArrayList<Object>();
93        row.add(s.getCode());
94        Map<String, Object> props = s.getMetadata();
95        String secName = "";
96        if (props.get("Q_SECONDARY_NAME") != null)
97          secName = (String) props.get("Q_SECONDARY_NAME");
98        row.add(secName);
99  
100       String extID = "";
101       if (props.get("Q_EXTERNALDB_ID") != null)
102         extID = (String) props.get("Q_EXTERNALDB_ID");
103       row.add(extID);
104 
105       haveData.addItem(row.toArray(new Object[row.size()]), i);
106     }
107 
108     haveData.setPageLength(samples.size());
109 
110     layout.addComponent(haveData);
111     Button ok = new Button("Close");
112     ok.addClickListener(new ClickListener() {
113 
114       @Override
115       public void buttonClick(ClickEvent event) {
116         subWindow.close();
117       }
118     });
119 
120     layout.addComponent(ok);
121 
122     subWindow.setContent(layout);
123     // Center it in the browser window
124     subWindow.center();
125     subWindow.setModal(true);
126     subWindow.setIcon(FontAwesome.FILE);
127     subWindow.setResizable(false);
128 
129     getUI().addWindow(subWindow);
130   }
