44  /**
45   * Wizard Step to life.qbic.projectwizard.model the biological entities of an experiment
46   * 
47   * @author Andreas Friedrich
48   * 
49   */
50  public class EntityStep implements WizardStep {
51  
52    private boolean skip = false;
53    private OptionGroup conditionsSet = new OptionGroup("dummy");
54  
55    private VerticalLayout main;
56  
57    private TextField expName;
58    private ComboBox species;
59    private TextField specialSpecies;
60    private ConditionsPanel cPanel;
61    private CheckBox infectionBox;
62  
63    private String emptyFactor = "Other (please specify)";
64    private List<String> suggestions = new ArrayList<String>(Arrays.asList("Age", "Genotype",
65        "Health State", "Phenotype", "Species", "Transient expression", "Treatment", emptyFactor));
66  
67    private OpenbisInfoTextField speciesNum;
68    private ComboBox person;
69    private Button reloadPeople;
70  
71    private OpenbisInfoTextField bioReps;
72  
73    public ConditionsPanel getCondPanel() {
74      return cPanel;
75    }
76  
77    public OptionGroup isConditionsSet() {
78      return conditionsSet;
79    }
80  
81    /**
82     * Create a new Entity step for the wizard
83     * 
84     * @param speciesMap A map of available species (codes and labels)
85     * @param people
86     */
87    public EntityStep(Map<String, String> speciesMap, Set<String> people) {
88      main = new VerticalLayout();
89      main.setMargin(true);
90      main.setSpacing(true);
91      Label header = new Label("Sample Sources");
92      main.addComponent(Styles.questionize(header,
93          "Sample sources are individual patients, animals or plants that are used in the experiment. "
94              + "You can input (optional) experimental variables, e.g. genotypes, that differ between different experimental groups.",
95          "Sample Sources"));
96      ArrayList<String> openbisSpecies = new ArrayList<String>();
97      openbisSpecies.addAll(speciesMap.keySet());
98      Collections.sort(openbisSpecies);
99      species = new ComboBox("Species", openbisSpecies);
100     species.setStyleName(Styles.boxTheme);
101     species.setRequired(true);
102     species.setFilteringMode(FilteringMode.CONTAINS);
103 
104     speciesNum = new OpenbisInfoTextField("How many different species are there in this project?",
105         "", "50px", "2");
106 
107     // new
108     VerticalLayout innerBox = new VerticalLayout();
109 
110     HorizontalLayout specialSpeciesOptions = Styles.questionize(innerBox,
111         "In an infection study one or more samples contain material from a different species. Please input the number of different organisms including the infecting species.",
112         "Number of Species");
113 
114     innerBox.setSpacing(true);
115     // specialSpeciesOptions.setMargin(true);
116     innerBox.addComponent(speciesNum.getInnerComponent());
117     infectionBox = new CheckBox("Infection Study");
118     innerBox.addComponent(infectionBox);
119     main.addComponent(specialSpeciesOptions);
120     // speciesNum.getInnerComponent().setVisible(false);
121     // speciesNum.getInnerComponent().setEnabled(false);
122     specialSpeciesOptions.setVisible(false);
123 
124     cPanel = new ConditionsPanel(suggestions, emptyFactor, "Species", species, true, conditionsSet,
125         specialSpeciesOptions);
126     main.addComponent(cPanel);
127 
128     expName = new TextField("Experimental Step Name");
129     expName.setStyleName(Styles.fieldTheme);
130     main.addComponent(expName);
131 
132     main.addComponent(specialSpeciesOptions);
133     main.addComponent(species);
134 
135     species.addValueChangeListener(new ValueChangeListener() {
136 
137       /**
138        * 
139        */
140       private static final long serialVersionUID = 1987640360028444299L;
141 
142       @Override
143       public void valueChange(ValueChangeEvent event) {
144         specialSpecies.setVisible(species.getValue().equals("Other"));
145       }
146     });
147     specialSpecies = new TextField("Species Information");
148     specialSpecies.setStyleName(Styles.fieldTheme);
149     specialSpecies.setVisible(false);
150     specialSpecies.setWidth("350px");
151     main.addComponent(specialSpecies);
152 
153     HorizontalLayout persBoxH = new HorizontalLayout();
154     persBoxH.setCaption("Contact Person");
155     person = new ComboBox();
156     person.addItems(people);
157     person.setFilteringMode(FilteringMode.CONTAINS);
158     person.setStyleName(Styles.boxTheme);
159 
160     reloadPeople = new Button();
161     Styles.iconButton(reloadPeople, FontAwesome.REFRESH);
162     persBoxH.addComponent(person);
163     persBoxH.addComponent(reloadPeople);
164 
165     main.addComponent(Styles.questionize(persBoxH,
166         "Contact person responsible for patients or sample sources.", "Contact Person"));
167 
168     bioReps = new OpenbisInfoTextField(
169         "How many identical biological replicates (e.g. animals) per group are there?",
170         "Number of (biological) replicates for each group."
171             + "Technical replicates are added later!",
172         "50px", "1");
173     main.addComponent(bioReps.getInnerComponent());
174   }
175 
176   public TextField getExpNameField() {
177     return expName;
178   }
179 
180   @Override
181   public String getCaption() {
182     return "Sample Sources";
183   }
184 
185   @Override
186   public Component getContent() {
187     return main;
188   }
189 
190   @Override
191   public boolean onAdvance() {
192     if (skip || speciesReady() && replicatesReady() && cPanel.isValid())
193       return true;
194     else {
195       Styles.notification("Missing information", "Please fill in the required fields.",
196           NotificationType.ERROR);
197       return false;
198     }
199   }
200 
201   private boolean replicatesReady() {
202     return !bioReps.getValue().isEmpty();
203   }
204 
205   private boolean speciesReady() {
206     return speciesIsFactor()
207         || (species.getValue() != null && !species.getValue().toString().isEmpty());
208   }
209 
210   @Override
211   public boolean onBack() {
212     return true;
213   }
214 
215   public boolean speciesIsFactor() {
216     return !species.isVisible();
217   }
218 
219   public void enableSpeciesFactor(boolean enable) {
220     speciesNum.getInnerComponent().setEnabled(enable);
221     speciesNum.getInnerComponent().setVisible(enable);
222     if (enable)
223       species.setValue(null);
224   }
225 
226   public List<String> getFactors() {
227     return cPanel.getConditions();
228   }
229 
230   public int getBioRepAmount() {
231     return Integer.parseInt(bioReps.getValue());
232   }
233 
234   public String getSpecies() {
235     if (species.getValue() == null)
236       return null;
237     else
238       return species.getValue().toString();
239   }
240 
241   public String getSpecialSpecies() {
242     return specialSpecies.getValue();
243   }
244 
245   public boolean factorFieldOther(ComboBox source) {
246     return emptyFactor.equals(source.getValue());
247   }
248 
249   public int getSpeciesAmount() {
250     return Integer.parseInt(speciesNum.getValue());
251   }
252 
253   public boolean isInfectionStudy() {
254     return infectionBox.getValue();
255   }
256 
257   public void setSkipStep(boolean b) {
258     skip = b;
259   }
260 
261   public boolean isSkipped() {
262     return skip;
263   }
264 
265   public String getPerson() {
266     if (person.getValue() != null)
267       return person.getValue().toString();
268     else
269       return null;
270   }
271 
272   public Button getPeopleReloadButton() {
273     return reloadPeople;
274   }
275 
276   public void updatePeople(Set<String> people) {
277     String contact = getPerson();
278     person.removeAllItems();
279     person.addItems(people);
280     if (contact != null && !contact.isEmpty())
281       person.select(contact);
282   }
