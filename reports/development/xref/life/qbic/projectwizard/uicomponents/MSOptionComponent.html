53    public MSOptionComponent(Vocabularies vocabs) {
54      this.setCaption("MS Experiment Options");
55      setSpacing(true);
56      proteinPooling = new CheckBox("Pool Before Protein Fractionation/Enrichment");
57      measurePeptides = new CheckBox("Measure Peptides");
58      shortGel = new CheckBox("Use Short Gel");
59      purification = new CheckBox("Protein Purification");
60  
61      addComponent(purification);
62      purificationMethods = new ComboBox("Purification Method");
63      purificationMethods.setNullSelectionAllowed(false);
64      purificationMethods.setStyleName(Styles.boxTheme);
65      purificationMethods.setVisible(false);
66      List<String> methods =
67          new ArrayList<String>(vocabs.getProteinPurificationMethodsMap().values());
68      Collections.sort(methods);
69      purificationMethods.addItems(methods);
70      addComponent(purificationMethods);
71  
72      purification.addValueChangeListener(new ValueChangeListener() {
73  
74        @Override
75        public void valueChange(ValueChangeEvent event) {
76          purificationMethods.setVisible(purification.getValue());
77        }
78      });
79      addComponent(shortGel);
80      addComponent(proteinPooling);
81      addComponent(measurePeptides);
82    }

85      boolean poolProteins = proteinPooling.getValue();
86      boolean peps = measurePeptides.getValue();
87      List<WizardStep> res = new ArrayList<WizardStep>();
88      if (poolProteins) {
89        res.add(steps.get(Steps.Test_Sample_Pooling));
90      }

92      res.add(f1);
93      if (peps) {
94        MSAnalyteStep../../../life/qbic/projectwizard/steps/MSAnalyteStep.html#MSAnalyteStep">MSAnalyteStep f2 = (MSAnalyteStep) steps.get(Steps.Peptide_Fractionation);
95        f1.setNeedsDigestion(true);
96        res.add(f2);

98        f1.setNeedsDigestion(false);

101     return res;
102   }

132

137   public void selectProteinPurification(String option) {
138     if(option.isEmpty()) {
139       purification.setValue(false);
140       purificationMethods.setNullSelectionAllowed(true);
141       purificationMethods.setValue(purificationMethods.getNullSelectionItemId());
142       purificationMethods.setNullSelectionAllowed(false);
143     } else {
144       purification.setValue(true);
145       purificationMethods.setValue(option);
146     }
147   }
