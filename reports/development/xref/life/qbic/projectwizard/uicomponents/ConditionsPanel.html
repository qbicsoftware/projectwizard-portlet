119   private void initListener() {
120     conditionChangeListener = new ValueChangeListener() {
121 
122       private static final long serialVersionUID = 1521218807043139513L;
123 
124       @Override
125       public void valueChange(ValueChangeEvent event) {
126         ComboBox source = (ComboBox) event.getProperty();
127         boolean special = false;
128         conditionsSet.setValue(null);
129         for (ConditionChooser c : choosers) {
130           if (c.getBox().equals(source)) {
131             c.changed();
132             if (c.chooserSet())
133               conditionsSet.select("set");
134           }
135           special |= c.factorIsSpecial();
136         }
137         specialField.setRequired(!special);
138         //TODO test this, was enabled
139         specialField.setVisible(!special);
140         specialOptionsComponent.setEnabled(special);
141         specialOptionsComponent.setVisible(special);
142       }
143     };
144 
145     buttonListener = new Button.ClickListener() {
146 
147       private static final long serialVersionUID = 2240224129259577437L;
148 
149       @Override
150       public void buttonClick(ClickEvent event) {
151         if (event.getButton().equals(add))
152           add();
153         else
154           remove();
155       }
156 
157     };
158   }
159 
160   public List<String> getConditions() {
161     List<String> res = new ArrayList<String>();
162     for (ConditionChooser c : choosers) {
163       if (c.isSet())
164         res.add(c.getCondition());
165     }
166     return res;
167   }
168 
169   public boolean isValid() {
170     boolean res = true;
171     for (ConditionChooser c : choosers) {
172       res &= c.isValid();
173     }
174     return res;
175   }
176 
177   public void changed() {
178     for (ConditionChooser c : choosers)
179       c.changed();
180   }
181 
182   private void add() {
183     ConditionChoosernts/ConditionChooser.html#ConditionChooser">ConditionChooser c = new ConditionChooser(options, other, special, nullSelectionAllowed);
184     c.addListener(conditionChangeListener);
185     choosers.add(c);
186 
187     removeComponent(buttonGridComp);
188     addComponent(c);
189     addComponent(buttonGridComp);
190   }
191 
192   private void remove() {
193     int size = choosers.size();
194     if (size > 1) {
195       ConditionChooser last = choosers.get(size - 1);
196       last.reset();
197       removeComponent(last);
198       choosers.remove(last);
199     }
200   }
