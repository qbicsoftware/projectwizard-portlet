18  package life.qbic.projectwizard.uicomponents;
19  
20  import java.util.List;
21  
22  public class LabelingMethod {
23  
24    private String name;
25    private List<String> reagents;
26  
27    public LabelingMethod(String name, List<String> reagents) {
28      this.name = name;
29      this.reagents = reagents;
30    }
31  
32    public String getName() {
33      return name;
34    }
35  
36    public List<String> getReagents() {
37      return reagents;
38    }
39  
40    @Override
41    public String toString() {
42      return name;
43    }
44  
45  }
