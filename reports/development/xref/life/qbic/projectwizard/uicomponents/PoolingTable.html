67    public PoolingTable(String name, TabSheet selectionTables, Map<Integer, Integer> usedTimes,
68        List<String> factorLabels) {
69      this.selectionTables = selectionTables;
70      all = (Table) selectionTables.getTab(0).getComponent();
71      used = (Table) selectionTables.getTab(1).getComponent();
72      labels = factorLabels;
73  
74      this.usedTimes = usedTimes;
75      setSpacing(true);
76  
77      HorizontalLayout tableButtonComponent = new HorizontalLayout();
78      tableButtonComponent.setCaption("Sample Pool");
79      tableButtonComponent.setSpacing(true);
80      secondaryName = new StandardTextField("Secondary Name");
81      secondaryName.setValue(name);
82      secondaryName.setStyleName(Styles.fieldTheme);
83      moveLeft = new Button();
84      Styles.iconButton(moveLeft, FontAwesome.ARROW_CIRCLE_LEFT);
85      
86      moveLeft.addStyleName("large_font_awesome");
87      addComponent(secondaryName);
88      poolIDs = new HashSet<Integer>();
89      poolTable = new Table();
90      initTable();
91      tableButtonComponent.addComponent(poolTable);
92      tableButtonComponent.addComponent(moveLeft);
93      addComponent(Styles
94          .questionize(
95              tableButtonComponent,
96              "You can add samples to the active pool by "
97                  + "selecting them from the right and clicking "+FontAwesome.ARROW_CIRCLE_LEFT.getHtml()+" or by dragging them over with your mouse.",
98              "Adding Samples to Pools"));
99  
100     initDragAndDrop(new Or(new SourceIs(all), new SourceIs(used)));
101     initButtonMover(all, used);
102   }

104   private Table getActiveTable() {
105     return (Table) selectionTables.getSelectedTab();
106   }

108   private boolean moveSampleToPool(Table source, Object itemId) {
109     boolean added = false;
110     final Integer id = (Integer) source.getItem(itemId).getItemProperty("ID").getValue();
111     String name = (String) source.getItem(itemId).getItemProperty("Secondary Name").getValue();
112     String labID = (String) source.getItem(itemId).getItemProperty("Lab ID").getValue();
113     final List<Object> row = new ArrayList<Object>();
114     row.add(id);
115     row.add(name);
116     row.add(labID);
117     for (String label : labels) {
118       String value = (String) source.getItem(itemId).getItemProperty(label).getValue();
119       row.add(value);
120     }
121     if (!poolIDs.contains(id)) {
122       added = true;
123       poolIDs.add(id);
124       Button delete = new Button();
125       Styles.iconButton(delete, FontAwesome.UNDO);
126       delete.setData(itemId);
127       delete.addClickListener(new Button.ClickListener() {
128         /**
129        * 
130        */
131         private static final long serialVersionUID = 5414603256990177472L;
132 
133         @Override
134         public void buttonClick(ClickEvent event) {
135           Integer iid = (Integer) event.getButton().getData();
136           poolIDs.remove(id);
137           poolTable.removeItem(iid);
138           resizeTable();
139           int newNumUsed = usedTimes.get(id) - 1;
140           if (newNumUsed == 0) {
141             usedTimes.remove(id);
142             used.removeItem(iid);
143             all.addItem(row.toArray(), iid);
144             all.sort(new Object[] {"ID"}, new boolean[] {true});
145           } else
146             usedTimes.put(id, newNumUsed);
147         }
148       });
149       poolTable.addItem(new Object[] {id, name, labID, delete}, itemId);
150       if (usedTimes.containsKey(id)) {
151         int newNumUsed = usedTimes.get(id) + 1;
152         usedTimes.put(id, newNumUsed);
153       } else {
154         usedTimes.put(id, 1);
155         all.removeItem(itemId);
156         used.addItem(row.toArray(), itemId);
157         used.sort(new Object[] {"ID"}, new boolean[] {true});
158       }
159     }
160     return added;
161   }

163   private void initButtonMover(Table sourceTable, Table usedTable) {
164     moveLeft.addClickListener(new Button.ClickListener() {
165 
166       /**
167        * 
168        */
169       private static final long serialVersionUID = 1L;
170 
171       @Override
172       public void buttonClick(ClickEvent event) {
173         Table source = getActiveTable();
174         Set<Object> ids = (Set<Object>) source.getValue();
175         boolean atLeastOneNew = false;
176         if (ids.size() > 0) {
177           for (Object itemId : ids) {
178             atLeastOneNew |= moveSampleToPool(source, itemId);
179           }
180           resizeTable();
181           if (!atLeastOneNew) {
182             Notification n = new Notification("Samples are already in this pool.");
183             n.setDelayMsec(3000);
184             n.show(Page.getCurrent());
185           }
186         }
187       }
188     });
189   }

191   private void initTable() {
192     poolTable.setStyleName(Styles.tableTheme);
193     resizeTable();
194     poolTable.addContainerProperty("ID", Integer.class, null);
195     poolTable.addContainerProperty("Secondary Name", String.class, null);
196     poolTable.addContainerProperty("Lab ID", String.class, null);
197     poolTable.addContainerProperty("Undo", Button.class, null);
198   }
199 
200   private void initDragAndDrop(final ClientSideCriterion acceptCriterion) {
201     poolTable.setDropHandler(new DropHandler() {
202       /**
203        * 
204        */
205       private static final long serialVersionUID = 4057757595986300434L;
206 
207       @Override
208       public void drop(final DragAndDropEvent dropEvent) {
209         // criteria verify that this is safe
210         final DataBoundTransferable t = (DataBoundTransferable) dropEvent.getTransferable();
211         if (!(t.getSourceComponent() instanceof Table)) {
212           return;
213         }
214 
215         Table source = (Table) t.getSourceComponent();
216         Object sourceItemId = t.getItemId();
217         if (!moveSampleToPool(source, sourceItemId)) {
218           Notification n = new Notification("Sample is already in this pool.");
219           n.setDelayMsec(3000);
220           n.show(Page.getCurrent());
221         }
222         resizeTable();
223       }
224 
225       @Override
226       public AcceptCriterion getAcceptCriterion() {
227         return new And(acceptCriterion, AcceptItem.ALL);
228       }
229     });
230   }
231 
232   private void resizeTable() {
233     poolTable.setPageLength(poolTable.size() + 1);
234   }
