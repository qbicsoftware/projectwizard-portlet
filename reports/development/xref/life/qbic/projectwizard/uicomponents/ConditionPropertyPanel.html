40  /**
41   * Composite UI component to input values of Property instances and their units
42   * 
43   * @author Andreas Friedrich
44   * 
45   */
46  public class ConditionPropertyPanel extends VerticalLayout {
47  
48    private static final long serialVersionUID = 3320102983685470217L;
49    private String condition;
50    private OptionGroup type;
51    private TextArea values;
52    private ComboBox unitField;
53    private static final Logger logger = LogManager.getLogger(ConditionPropertyPanel.class);
54  
55    /**
56     * Create a new Condition Property Panel
57     * 
58     * @param condition The name of the condition selected
59     * @param units An EnumSet of units (e.g. SI units)
60     */
61    public ConditionPropertyPanel(String condition, EnumSet<life.qbic.xml.properties.Unit> units) {
62      this.condition = condition;
63      type = new OptionGroup("", new ArrayList<String>(Arrays.asList("Continuous", "Categorical")));
64      // type = new CustomVisibilityComponent(new OptionGroup("", new
65      // ArrayList<String>(Arrays.asList("Continuous", "Categorical"))));
66  
67      values = new TextArea("Values");
68      values.setWidth("300px");
69      values.setInputPrompt("Please input different occurrences of the condition " + condition + ",\n"
70          + "one per row.");
71      Validator v = new PropertyValuesValidator();
72      values.addValidator(v);
73      values.setValidationVisible(true);
74      values.setImmediate(true);
75      values.setRequired(true);
76      values.setRequiredError("Please input at least one condition.");
77  
78      unitField = new ComboBox("Unit", units);
79      unitField.setStyleName(Styles.boxTheme);
80      unitField.setEnabled(false);
81      unitField.setVisible(false);
82      unitField.setNullSelectionAllowed(false);
83      NullValidator uv =
84          new NullValidator("If the condition is continuous, a unit must be selected.", false);
85      unitField.addValidator(uv);
86      unitField.setValidationVisible(false);
87  
88      initListener();
89      addComponent(values);
90      addComponent(Styles.questionize(type,
91          "Continuous variables can be measured and have units, "
92              + "e.g. discrete time points, categorical variables don't, e.g. different genotypes.",
93          "Variable Type"));
94      addComponent(unitField);
95      setSpacing(true);
96    }
97  
98    private void initListener() {
99      ValueChangeListener typeListener = new ValueChangeListener() {
100 
101       /**
102        * 
103        */
104       private static final long serialVersionUID = -6989982426500636013L;
105 
106       @Override
107       public void valueChange(ValueChangeEvent event) {
108         boolean on = type != null && type.getValue().toString().equals("Continuous");
109         unitField.setEnabled(on);
110         unitField.setVisible(on);
111         unitField.setValidationVisible(on);
112         if (!on)
113           unitField.select(unitField.getNullSelectionItemId());
114       }
115     };
116     type.addValueChangeListener(typeListener);
117   }
118 
119   /**
120    * Returns all conditions with their units as a list
121    * 
122    * @return
123    */
124   public List<Property> getFactors() {
125     List<Property> res = new ArrayList<Property>();
126     life.qbic.xml.properties.Unit unit = null;
127     if (unitField.getValue() != null)
128       unit = (life.qbic.xml.properties.Unit) unitField.getValue();
129     for (String val : values.getValue().split("\n")) {
130       val = val.trim();
131       if (unit != null)
132         res.add(new Property(condition.toLowerCase().replace(" ", "_"), val, unit,
133             PropertyType.Factor));
134       else
135         res.add(new Property(condition.toLowerCase().replace(" ", "_"), val, PropertyType.Factor));
136     }
137     return res;
138   }
139 
140   public boolean unitSet() {
141     return unitField.getValue() != null;
142   }
143 
144   public boolean isContinuous() {
145     return type.getValue() != null && type.getValue().toString().equals("Continuous");
146   }
147 
148   public ComboBox getUnitsBox() {
149     return unitField;
150   }
151 
152   public TextArea getInputArea() {
153     return values;
154   }
155 
156   public OptionGroup getUnitTypeSelect() {
157     return type;
158   }
159 
160   public boolean isValid() {
161     return values.isValid();
162   }
163 
164   public String getException() {
165     String err = "";
166     String val = values.getValue();
167     if (val.isEmpty())
168       err = "Please input different occurrences of the condition " + condition + ",\n"
169           + "one per row.";
170     else if (val.contains(";") || val.contains(":"))
171       err = "Please don't use the symbols ';' or ':'";
172     else if (isContinuous() && !unitSet())
173       err = "If the condition is continuous, a unit must be selected.";
174     return err;
175   }
176 
177   public class PropertyValuesValidator implements Validator {
178 
179     @Override
180     public void validate(Object value) throws InvalidValueException {
181       String val = (String) value;
182       if (val != null && !val.isEmpty())
183         // throw new InvalidValueException("Please fill in values.");
184         if (val.contains(":") || val.contains(";"))
185         throw new InvalidValueException("Please don't use the symbols ';' or ':'");
186     }
187   }
188 }
