452   public boolean isValid() {
453     boolean res = true;
454     String error = "";
455     for (Iterator i = sampleTable.getItemIds().iterator(); i.hasNext();) {
456       // Get the current item identifier, which is an integer.
457       int iid = (Integer) i.next();
458 
459       // Now get the actual item from the table.
460       // Item item = proteinExperiments.getItem(iid);
461       error = "Please fill in the number of " + type + "s (or '0' for none)"; // TODO old?
462 
463       String fractions = parseTextRow(iid, type + "s").getValue();
464       if (!fractions.isEmpty()) {
465         try {
466           Integer.parseInt(fractions);
467         } catch (NumberFormatException e) {
468           res = false;
469         }
470       } else {
471         res = false;
472       }
473     }
474 
475     if (!res) {
476       Styles.notification("Missing Input", error, NotificationType.ERROR);
477       return false;
478     } else
479       return true;
480   }

494   public MSExperimentModelizard/model/MSExperimentModel.html#MSExperimentModel">MSExperimentModel getFractionsWithMSProperties(MSExperimentModel model, String sampleType,
495       String method) {
496     Map<String, ExperimentModel> fractHelper = new HashMap<String, ExperimentModel>();
497     List<ExperimentModel> fractionations = new ArrayList<ExperimentModel>();
498     List<ExperimentModel> msExperiments = new ArrayList<ExperimentModel>();
499     List<ExperimentModel> peptides = new ArrayList<ExperimentModel>();
500 
501     Map<String, Object> props = generalFractionMSInfo.getExperimentalProperties();
502     // there can be one ms measurement experiment per protein/peptide sample measured (fractions)
503     // new ms experiment
504     ExperimentModel msExp = new ExperimentModel(0, new ArrayList<AOpenbisSample>());
505     msExp.setProperties(props);
506 
507     Map<String, List<AOpenbisSample>> peptidesPerDigestion =
508         new HashMap<String, List<AOpenbisSample>>();
509     // collect samples
510     tableIdToSample = new HashMap<Object, AOpenbisSample>();
511     for (Object i : sampleTable.getItemIds()) {
512       String item = (String) i;
513       String[] ids = item.split("-");
514 
515       AOpenbisSample parent = tableIdToParent.get(item);
516 
517       // new fraction/enrichment cycled sample - this always exists if it's in the table
518       String secondaryName = parseTextRow(i, type.toString() + " Name").getValue();
519       String extID = parseTextRow(i, type.toString() + " Lab ID").getValue();
520       OpenbisTestSample fractionSample =
521           new OpenbisTestSample(-2, new ArrayList<AOpenbisSample>(Arrays.asList(parent)),
522               sampleType, secondaryName, extID, new ArrayList<Property>(), "");
523 
524       ComboBox selection = parseBoxRow(i, "Process");
525       String option = selection.getValue().toString();
526       if (option.equals("Both") || option.equals("Measure") || sampleType.equals("PEPTIDES")) {
527         // new ms sample
528         OpenbisMSSample msSample =
529             new OpenbisMSSample(-1, new ArrayList<AOpenbisSample>(Arrays.asList(fractionSample)),
530                 secondaryName + " run", extID + " run", new ArrayList<Property>(), "");
531         msExp.addSample(msSample);
532         // used to attach wash samples
533         tableIdToSample.put(i, msSample);
534         // if we have at least one ms measurement, the experiment gets added to the experiment list
535         if (msExperiments.isEmpty())
536           msExperiments.add(msExp);
537       }
538       if (option.equals("Both") || option.equals("Digest")) {
539         OpenbisTestSample pepSample =
540             new OpenbisTestSample(-1, new ArrayList<AOpenbisSample>(Arrays.asList(fractionSample)),
541                 "PEPTIDES", fractionSample.getQ_SECONDARY_NAME() + " digested",
542                 fractionSample.getQ_EXTERNALDB_ID(), new ArrayList<Property>(), "");
543         List<String> enzymes = getEnzymesFromSampleRow(i);
544         String digestion = StringUtils.join(enzymes, ", ");
545         if (peptidesPerDigestion.containsKey(digestion))
546           peptidesPerDigestion.get(digestion).add(pepSample);
547         else
548           peptidesPerDigestion.put(digestion,
549               new ArrayList<AOpenbisSample>(Arrays.asList(pepSample)));
550       }
551       if (fractHelper.containsKey(ids[0])) {
552         fractHelper.get(ids[0]).addSample(fractionSample);
553       } else {
554         // there is only one fractionation experiment per fractionation containing all of the
555         // resulting samples
556         ExperimentModel fractionExp =
557             new ExperimentModel(item, new ArrayList<AOpenbisSample>(Arrays.asList(fractionSample)));
558         switch (type) {
559           case Fraction:
560             fractionExp.addProperty("Q_MS_FRACTIONATION_METHOD", method);
561             break;
562           case Cycle:
563             fractionExp.addProperty("Q_MS_ENRICHMENT_METHOD", method);
564             break;
565           default:
566             logger.error("Unknown AnalyteMultiplicationType: " + type);
567             break;
568         }
569         fractHelper.put(ids[0], fractionExp);
570         if (sampleType.equals("PEPTIDES"))
571           peptides.add(fractionExp);
572         else
573           fractionations.add(fractionExp);
574       }
575     }
576     // one digestion experiment per unique enzyme set used
577     for (String digestion : peptidesPerDigestion.keySet()) {
578       ExperimentModel peptideExp = new ExperimentModel("", peptidesPerDigestion.get(digestion));
579       peptideExp.addProperty("Q_ADDITIONAL_INFO", "Digestion: " + digestion);
580       peptides.add(peptideExp);
581     }
582 
583     if (fractionations.size() > 0)
584       model.addAnalyteStepExperiments(fractionations);
585     if (msExperiments.size() > 0)
586       model.addMSRunStepExperiments(msExperiments);
587     if (peptides.size() > 0)
588       model.addDigestionExperiment(peptides);
589     if (poolSamples.getValue()) {
590       List<AOpenbisSample> pools = new ArrayList<AOpenbisSample>();
591       int i = 0;
592       for (String id : fractHelper.keySet()) {
593         i++;
594 
595         List<AOpenbisSample> fractions = fractHelper.get(id).getSamples();
596         AOpenbisSample pool = new OpenbisTestSample(1, fractions, sampleType,
597             sampleType.toLowerCase() + " pool " + Integer.toString(i), "", new ArrayList<Property>(),
598             "");
599         pools.add(pool);
600 
601         if (sampleType.equals("PEPTIDES")) {
602           ExperimentModel msExpPool =
603               new ExperimentModel(2, new ArrayList<AOpenbisSample>(Arrays.asList(pool)));
604           msExp.setProperties(props);
605           model.getLastStepMsRuns().add(msExpPool);
606         } else {
607           OpenbisTestSample pepSample =
608               new OpenbisTestSample(-1, new ArrayList<AOpenbisSample>(Arrays.asList(pool)),
609                   "PEPTIDES", pool.getQ_SECONDARY_NAME() + " digested", pool.getQ_EXTERNALDB_ID(),
610                   new ArrayList<Property>(), "");
611           ExperimentModel peptideExpPool =
612               new ExperimentModel(1, new ArrayList<AOpenbisSample>(Arrays.asList(pepSample)));
613           model.getPeptideExperiments().add(peptideExpPool);
614         }
615       }
616 
617       if (sampleType.equals("PEPTIDES")) {
618         model.getPeptideExperiments().add(new ExperimentModel(3, pools));
619       } else {
620         model.getLastStepAnalytes().add(new ExperimentModel(3, pools));
621       }
622     }
623     return model;
624   }
