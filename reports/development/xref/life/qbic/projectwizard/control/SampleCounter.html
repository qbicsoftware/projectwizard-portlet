16  package life.qbic.projectwizard.control;
17  
18  import java.util.ArrayList;
19  import java.util.Arrays;
20  import java.util.HashSet;
21  import java.util.List;
22  import java.util.Set;
23  import org.apache.logging.log4j.LogManager;
24  import org.apache.logging.log4j.Logger;
25  import ch.ethz.sis.openbis.generic.asapi.v3.dto.sample.Sample;
26  import life.qbic.datamodel.identifiers.SampleCodeFunctions;
27  
28  public class SampleCounter {
29  
30    private int entityID;
31    // private int barcodeID;
32    private int expID;
33    private String barcode;
34    private String project;
35    Logger logger = LogManager.getLogger(SampleCounter.class);
36    private Set<String> knownAtypical = new HashSet<>();
37  
38    public SampleCounter(List<Sample> samples) {
39      this(samples.get(0).getCode().substring(0, 5));
40      for (Sample s : samples)
41        increment(s);
42    }
43  
44    public SampleCounter(String project) {
45      entityID = 0;
46      expID = 0;
47      barcode = "";
48      // barcodeID = 1;
49      this.project = project;
50      knownAtypical.add(project + "_INFO");
51    }
52  
53    // TODO later updates (after initialization)
54    public void increment(Sample sample) {
55      String code = sample.getCode();
56      String experimentID = null;
57      try {
58        experimentID = sample.getExperiment().getIdentifier().getIdentifier();
59      } catch (Exception e) {
60      }
61  
62      try {
63        String expNumString = experimentID.split(project + "E")[1];
64        int expNum = Integer.parseInt(expNumString);
65        if (expNum > expID)
66          expID = expNum;
67      } catch (Exception e) {
68        if (!knownAtypical.contains(experimentID)) {
69          knownAtypical.add(experimentID);
70          logger.warn("While counting existing experiments in project " + project
71              + " unfamiliar experiment identifier " + experimentID + " was found.");
72        }
73      }
74      if (SampleCodeFunctions.isQbicBarcode(code)) {
75        // previously registered samples...
76        List<String> specialCases = new ArrayList<String>(Arrays.asList("QMSHS001X3", "QMSHS002XB",
77            "QMSHS003XJ", "QMSHS004XR", "QMSHS005X1", "QMSHS006X9"));
78        if (code.startsWith("QMSHS999W"))
79          barcode = "QMSHS006X9";
80        if (!specialCases.contains(code)) {
81          if (SampleCodeFunctions.compareSampleCodes(code, barcode) > 0)
82            barcode = code;
83        }
84      } else if (sample.getType().getCode().equals(("Q_BIOLOGICAL_ENTITY"))) {
85        int num = Integer.parseInt(sample.getCode().split("-")[1]);
86        if (num >= entityID)
87          entityID = num;
88      }
89    }
90  
91    public String getNewExperiment() {
92      expID++;
93      return project + "E" + expID;
94    }
95  
96    public String getNewEntity() {
97      entityID++;
98      return project + "ENTITY-" + Integer.toString(entityID);
99    }
100 
101   public String getNewBarcode() {
102     if (barcode == null || barcode.isEmpty()) {
103       barcode = project + "001A";
104       barcode = barcode + SampleCodeFunctions.checksum(barcode);
105     }
106     barcode = SampleCodeFunctions.incrementSampleCode(barcode);
107     return barcode;
108   }
109 
110 }
