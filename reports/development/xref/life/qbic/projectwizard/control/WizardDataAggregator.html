173   /**
174    * Fetches context information like space and project and computes first unused IDs of samples and
175    * context.
176    */
177   private void prepareBasics() {
178     firstFreeExperimentID = 1;
179     firstFreeEntityID = 1;
180     firstFreeBarcode = "";// TODO cleanup where not needed
181     existingSamples = new HashMap<String, Sample>();
182     spaceCode = s1.getSpaceCode();
183     projectCode = s1.getProjectCode().toUpperCase();
184 
185     samples = new ArrayList<Sample>();
186     if (openbis.projectExists(spaceCode, projectCode)) {
187       samples.addAll(openbis.getSamplesOfProject("/" + spaceCode + "/" + projectCode));
188     }
189 
190     if (!s1.fetchTSVModeSet()) {
191       for (Experiment e : openbis.getExperimentsOfProjectByCode(projectCode)) {
192         String code = e.getCode();
193         String[] split = code.split(projectCode + "E");
194         if (code.startsWith(projectCode + "E") && split.length > 1) {
195           int num = 0;
196           try {
197             num = Integer.parseInt(split[1]);
198           } catch (Exception e2) {
199           }
200           if (firstFreeExperimentID <= num)
201             firstFreeExperimentID = num + 1;
202         }
203       }
204 
205       for (Sample s : samples) {
206         String code = s.getCode();
207         existingSamples.put(code, s);
208         if (SampleCodeFunctions.isQbicBarcode(code)) {
209           if (SampleCodeFunctions.compareSampleCodes(firstFreeBarcode, code) <= 0) {
210             firstFreeBarcode = SampleCodeFunctions.incrementSampleCode(code);
211           }
212         } else if (s.getType().getCode().equals(("Q_BIOLOGICAL_ENTITY"))) {
213           int num = Integer.parseInt(s.getCode().split("-")[1]);
214           if (num >= firstFreeEntityID)
215             firstFreeEntityID = num + 1;
216         }
217       }
218     }
219   }

222    * Creates the list of biological entities from the input information collected in the aggregator
223    * fields and wizard steps and fetches or creates the associated context
224    * 
225    * @param map

227    * @return
228    * @throws JAXBException

230   public List<AOpenbisSample> prepareEntities(Map<Object, Integer> map, boolean copy,
231       boolean infectionStudy) throws JAXBException {
232     prepareBasics();
233     this.factorMap = new HashMap<String, Property>();
234     experiments = new ArrayList<OpenbisExperiment>();
235     species = s2.getSpecies();
236     speciesSpecified = s2.getSpecialSpecies();
237     bioReps = s2.getBioRepAmount();
238 
239     // entities are not created new, but parsed from registered ones
240     if (inheritEntities) {
241       openbisEntities = openbis.getSamplesofExperiment(s1.getExperiment().getID());
242       entities = parseEntities(openbisEntities, copy);
243       // create new entities and an associated experiment from collected inputs
244     } else {
245       int personID = -1;
246       String person = s2.getPerson();
247       Map<String, Object> props = new HashMap<String, Object>();
248       if (!s2.getExpNameField().getValue().isEmpty())
249         props.put("Q_SECONDARY_NAME", s2.getExpNameField().getValue());
250       if (person != null && !person.isEmpty())
251         personID = personMap.get(person);
252       experiments.add(new OpenbisExperiment(buildExperimentName(),
253           ExperimentType.Q_EXPERIMENTAL_DESIGN, personID, props));
254 
255       List<List<Property>> valueLists = s3.getFactors();
256       bioFactors = createFactorInfo(valueLists);
257 
258       entities = buildEntities(map, infectionStudy);
259     }
260     return entities;
261   }

263   /**
264    * Creates the list of biological extracts from the input information collected in the aggregator
265    * fields and wizard steps and fetches or creates the associated context
266    * 
267    * @param map
268    * @param copyMode
269    * 
270    * @return
271    * @throws JAXBException
272    */
273   public List<AOpenbisSample> prepareExtracts(Map<Object, Integer> map, boolean copyMode)
274       throws JAXBException {
275     // extracts are not created new, but parsed from registered ones
276     if (inheritExtracts) {
277       String expID = "";
278       List<Sample> samples = new ArrayList<>();
279       if (copyMode) {
280         // child experiment of entities
281         expID = existingSamples.get(openbisEntities.get(0).getCode()).getChildren().get(0)
282             .getExperiment().getIdentifier().getIdentifier();
283         samples = openbis.getSamplesofExperiment(expID);
284       } else {
285         samples = openbis.getSamplesofExperiment(s1.getExperiment().getID());
286       }
287 
288       Map<Sample, List<String>> parentMap = new HashMap<>();
289       Map<Sample, List<Sample>> parentSampleMap = openbis.getParentMap(samples);
290       for (Sample key : parentSampleMap.keySet()) {
291         List<String> parents = new ArrayList<>();
292         for (Sample p : parentSampleMap.get(key)) {
293           parents.add(p.getCode());
294         }
295         parentMap.put(key, parents);
296       }
297       if (copyMode) {
298         List<AOpenbisSample> oldExtracts = parseExtracts(samples, parentMap);
299         Set<String> entityCodes = new HashSet<String>();
300         for (AOpenbisSample e : entities)
301           entityCodes.add(e.getCode());
302         extracts = new ArrayList<AOpenbisSample>();
303         for (AOpenbisSample s : oldExtracts) {
304           boolean orphan = true;
305           String parentString = s.getParent();
306           for (String p : parentString.split(" ")) {
307             String newP = oldCodesToNewCodes.get(p);
308             if (entityCodes.contains(newP)) {
309               orphan = false;
310               parentString = parentString.replace(p, newP);
311             } else {
312               parentString = parentString.replace(p, " ");
313             }
314           }
315           if (!orphan) {
316             incrementOrCreateBarcode();
317             String newCode = nextBarcode;
318             oldCodesToNewCodes.put(s.getCode(), newCode);
319             s.setCode(newCode);
320             s.setParent(parentString);
321             extracts.add(s);
322           }
323         }
324       } else {
325         prepareBasics();
326         this.factorMap = new HashMap<String, Property>();
327         experiments = new ArrayList<OpenbisExperiment>();
328 
329         extracts = parseExtracts(samples, parentMap);
330       }
331       // create new entities and an associated experiment from collected inputs
332     } else {
333       tissue = s5.getTissue();
334       specialTissue = s5.getCellLine();
335       if ("Other".equals(tissue))
336         specialTissue = s5.getSpecialTissue();
337       extractReps = s5.getExtractRepAmount();
338       int personID = -1;
339       String person = s5.getPerson();
340       if (person != null && !person.isEmpty())
341         personID = personMap.get(person);
342       Map<String, Object> props = new HashMap<String, Object>();
343       if (!s5.getExpNameField().getValue().isEmpty())
344         props.put("Q_SECONDARY_NAME", s5.getExpNameField().getValue());
345       experiments.add(new OpenbisExperiment(buildExperimentName(),
346           ExperimentType.Q_SAMPLE_EXTRACTION, personID, props));
347       List<List<Property>> valueLists = s6.getFactors();
348       extractFactors = createFactorInfo(valueLists);
349       // keep track of id letters for different conditions
350       classChars = new HashMap<String, Character>();
351       extracts = buildExtracts(entities, classChars, map);
352     }
353     return extracts;
354   }

358    * aggregator fields and wizard steps and fetches or creates the associated context
359    * 
360    * @return
361    */
362   public List<List<AOpenbisSample>> prepareTestSamples() {
363     techTypeInfo = s8.getAnalyteInformation();
364     if (inheritExtracts) {
365       prepareBasics();
366       classChars = new HashMap<String, Character>();
367       experiments = new ArrayList<OpenbisExperiment>();
368     }
369     for (TestSampleInformation x : techTypeInfo) {
370       int personID = -1;
371       String person = x.getPerson();
372       if (person != null && !person.isEmpty())
373         personID = personMap.get(person);
374       OpenbisExperiment exp = new OpenbisExperiment(buildExperimentName(),
375           ExperimentType.Q_SAMPLE_PREPARATION, personID, null);// TODO add secondary name here
376       experiments.add(exp);
377     }
378     List<List<AOpenbisSample>> techSortedTests = buildTestSamples(extracts, classChars);
379     tests = new ArrayList<AOpenbisSample>();
380     for (List<AOpenbisSample> group : techSortedTests)
381       tests.addAll(group);
382     for (int i = techSortedTests.size() - 1; i > -1; i--) {
383       if (!techTypeInfo.get(i).isPooled() && !s8.hasComplexProteinPoolBeforeFractionation())
384         techSortedTests.remove(i);
385     }
386     return techSortedTests;
387   }

417   /**
418    * Build and return a list of all possible MHC ligand extracts, using existing test samples and
419    * the number of antibody columns used in the experiment.
420    * 
421    * @param tests prepared ligands (test samples) these MHC ligand extracts will be prepared from
422    *        (and attached to)
423    * @param classChars Filled map of different class letters used for the tests
424    * @return List of lists of AOpenbisSamples containing MHC ligand extract samples
425    */
426   private List<AOpenbisSample> buildMHCExtractSamples(List<AOpenbisSample> tests,
427       Map<String, Character> classChars, String tissueCode, MHCLigandExtractionProtocol protocol) {
428     List<AOpenbisSample> mhcExtracts = new ArrayList<AOpenbisSample>();
429     int expNum = experiments.size() - 1;
430     for (AOpenbisSample s : tests) {
431       String type = s.getValueMap().get("Q_SAMPLE_TYPE");
432       if (type.equals("CELL_LYSATE")) {
433         if (s.getParent().equals(tissueCode)) {
434           String secondaryName = s.getQ_SECONDARY_NAME();
435           for (String antibody : protocol.getAntibodies()) {
436             String[] mhcClasses = protocol.getMHCClass(antibody);
437             for (String mhcClass : mhcClasses) {
438               if (classChars.containsKey(secondaryName)) { // TODO see other sample creation
439                 classChar = classChars.get(secondaryName);
440               } else {
441                 classChar = SampleCodeFunctions.incrementUppercase(classChar);
442                 classChars.put(secondaryName, classChar);
443               }
444               incrementOrCreateBarcode();
445               mhcExtracts.add(new OpenbisMHCExtractSample(nextBarcode, spaceCode,
446                   experiments.get(expNum).getExperimentCode(), secondaryName, "", s.getFactors(),
447                   antibody, mhcClass, s.getCode(), s.getQ_EXTERNALDB_ID()));
448             }
449           }
450         }
451       }
452     }
453     return mhcExtracts;
454   }

483   /**
484    * Collects conditions as Strings in a list of their instance lists. Also puts the conditions
485    * (factors) in a HashMap for later lookup, using value and unit as a key
486    * 
487    * @param factors List of a list of condition instances (one list per condition)
488    * @return List of a list of condition instances (one list per condition) as Strings
489    */
490   private List<List<String>> createFactorInfo(List<List<Property>> factors) {
491     List<List<String>> res = new ArrayList<List<String>>();
492     for (List<Property> instances : factors) {
493       List<String> factorValues = new ArrayList<String>();
494       for (Property f : instances) {
495         String name = f.getValue();
496         if (f.hasUnit())
497           name = name + " " + f.getUnit().getValue();
498         factorValues.add(name);
499         factorMap.put(name, f);
500       }
501       res.add(factorValues);
502     }
503     return res;
504   }

516   /**
517    * Generates all permutations of a list of experiment conditions
518    * 
519    * @param lists Instance lists of different conditions
520    * @param infectionStudy
521    * @return List of all possible permutations of the input conditions
522    */
523   public List<String> generatePermutations(List<List<String>> lists, boolean infectionStudy) {
524     infectantSpecies = new HashSet<>();
525     List<String> res = new ArrayList<>();
526     generatePermutationsHelper(lists, res, 0, "");
527 
528     if (infectionStudy) {
529       List<String> newRes = new ArrayList<>();
530       int n = 0;
531       for (String concat : res) {
532         String[] conditionSplit = concat.split("###");
533 
534         // handle infection studies. every second entry is a potential infecting species
535         n++;
536         if ((n % 2) == 0) {
537           for (String condition : conditionSplit) {
538             if (taxMap.containsKey(condition)) {
539               infectantSpecies.add(condition);
540             }
541           }
542         } else {
543           newRes.add(concat);
544         }
545       }
546       res = newRes;
547     }
548     return res;
549   }

551   /**
552    * recursive helper
553    */
554   private void generatePermutationsHelper(List<List<String>> lists, List<String> result, int depth,
555       String current) {
556     String separator = "###";
557     if (depth == lists.size()) {
558       result.add(current);
559       return;
560     }
561     for (int i = 0; i < lists.get(depth).size(); ++i) {
562       if (current.equals(""))
563         separator = "";
564       generatePermutationsHelper(lists, result, depth + 1,
565           current + separator + lists.get(depth).get(i));
566     }
567   }

569   /**
570    * Build and return a list of all possible biological entities given their conditions, keep track
571    * of conditions in a HashMap for later
572    * 
573    * @param map
574    * 
575    * @return List of AOpenbisSamples containing entity samples
576    */
577   private List<AOpenbisSample> buildEntities(Map<Object, Integer> map, boolean infectionStudy) {
578     List<AOpenbisSample> entities = new ArrayList<AOpenbisSample>();
579     List<List<String>> factorLists = new ArrayList<List<String>>();
580     factorLists.addAll(bioFactors);
581     List<String> permutations = generatePermutations(factorLists, infectionStudy);
582 
583     List<List<String>> permLists = new ArrayList<List<String>>();
584     for (String concat : permutations) {
585       String[] conditionSplit = concat.split("###");
586       permLists.add(new ArrayList<String>(Arrays.asList(conditionSplit)));
587     }
588 
589     int entityNum = firstFreeEntityID;
590     int defBioReps = bioReps;
591     int permID = 0;
592 
593     for (List<String> secondaryNameList : permLists) {
594       permID++;
595       String secondaryName = nameListToSecondaryName(secondaryNameList);
596       if (map.containsKey(permID))
597         defBioReps = map.get(permID);
598       for (int i = defBioReps; i > 0; i--) {
599         List<Property> factors = new ArrayList<Property>();
600         for (String name : secondaryNameList) {
601           if (factorMap.containsKey(name))
602             factors.add(factorMap.get(name));
603         }
604         if (s2.speciesIsFactor()) {
605           for (String factor : secondaryNameList) {
606             if (taxMap.containsKey(factor))
607               species = factor;
608           }
609         }
610         String taxID = taxMap.get(species);
611 
612         entities.add(new OpenbisBiologicalEntity(projectCode + "ENTITY-" + entityNum, spaceCode,
613             experiments.get(0).getExperimentCode(), secondaryName, "", factors, taxID, "", ""));
614         entityNum++;
615       }
616     }
617     backgroundEntities = new ArrayList<>();
618     if (infectionStudy) {
619       int personID = -1;
620       String person = s2.getPerson();
621       Map<String, Object> props = new HashMap<String, Object>();

640   /**
641    * Build and return a list of all possible biological extracts given their conditions, using
642    * existing entities. Keep track of condition in a HashMap for later
643    * 
644    * @param entities Existing (or prepared) biological entity samples these extracts will be
645    *        attached to
646    * @param classChars Empty map of different class letters used for the identifiers, to keep track
647    *        of for test samples
648    * @param map
649    * @return List of AOpenbisSamples containing extract samples
650    */
651   private List<AOpenbisSample> buildExtracts(List<AOpenbisSample> entities,
652       Map<String, Character> classChars, Map<Object, Integer> map) {
653     int expNum = experiments.size() - techTypeInfo.size() - 1;
654     List<AOpenbisSample> extracts = new ArrayList<AOpenbisSample>();
655     int permID = 0;
656     Map<String, TissueInfo> specialTissueInfos = s6.getSpecialTissueMap();
657     for (AOpenbisSample e : entities) {
658       List<List<String>> factorLists = new ArrayList<List<String>>();
659       String secName = e.getQ_SECONDARY_NAME();
660       if (secName == null)
661         secName = "";
662       factorLists.add(new ArrayList<String>(Arrays.asList(secName)));
663 
664       factorLists.addAll(extractFactors);
665       List<String> permutations = generatePermutations(factorLists, false);
666       List<List<String>> permLists = new ArrayList<List<String>>();
667       for (String concat : permutations) {
668         permLists.add(new ArrayList<String>(Arrays.asList(concat.split("###"))));
669       }
670       for (List<String> secondaryNameList : permLists) {
671         permID++;
672         List<Property> factors = new ArrayList<Property>();
673         factors.addAll(e.getFactors());
674         for (String name : secondaryNameList)
675           for (String element : name.split(";")) {
676             element = element.trim();
677             if (factorMap.containsKey(element)) {
678               if (!factors.contains(factorMap.get(element)))
679                 factors.add(factorMap.get(element));
680             }
681           }
682         String secondaryName = nameListToSecondaryName(secondaryNameList);
683         int defExtrReps = extractReps;
684         if (map.containsKey(permID))
685           defExtrReps = map.get(permID);
686         for (int i = defExtrReps; i > 0; i--) {
687           String tissueCode = "";
688           if (s5.isTissueFactor()) {
689             for (String factorInstance : secondaryNameList) {
690               if (specialTissueInfos.containsKey(factorInstance)) {
691                 TissueInfo info = specialTissueInfos.get(factorInstance);
692                 tissue = info.getPrimary();
693                 specialTissue = info.getSpecific();
694               }
695             }
696           }
697           tissueCode = tissueMap.get(tissue);
698           if (classChars.containsKey(secondaryName)) { // TODO does this seem right to you?
699             classChar = classChars.get(secondaryName);
700           } else {
701             classChar = SampleCodeFunctions.incrementUppercase(classChar);
702             classChars.put(secondaryName, classChar);
703           }
704           List<Property> curFactors = new ArrayList<Property>(factors);
705           incrementOrCreateBarcode();
706           extracts.add(new OpenbisBiologicalSample(nextBarcode, spaceCode,
707               experiments.get(expNum).getExperimentCode(), secondaryName, "", curFactors,
708               tissueCode, specialTissue, e.getCode(), e.getQ_EXTERNALDB_ID())); // TODO
709           // ext
710           // db
711           // id
712         }
713       }
714     }
715     return extracts;

1053     AOpenbisSample a = samples.get(0);

1061     String secondaryName = s1.getExpSecondaryName();
1062     String investigator = s1.getPerson(PersonType.Investigator);
1063     String contact = s1.getPerson(PersonType.Contact);

1067     description = description.replace("\n", "\n#");
1068     secondaryName = secondaryName.replace("\n", " - ");
1069     result += "#PROJECT_DESCRIPTION=" + description + "\n";
1070     result += "#ALTERNATIVE_NAME=" + secondaryName + "\n";
1071     if (s1.isPilot())
1072       result += "#PILOT PROJECT\n";
1073     result += "#INVESTIGATOR=" + investigator + "\n";
1074     result += "#CONTACT=" + contact + "\n";
1075     result += "#MANAGER=" + manager + "\n";
1076 
1077     List<OpenbisExperiment> allExperiments = new ArrayList<>();
1078     if (experiments != null) {
1079       allExperiments.addAll(experiments);
1080     }

1092     // if (msProps != null)
1093     // result += addExperimentInfoLine(msProps, "Q_MS_MEASUREMENT") + "\n";
1094     if (mhcExperimentProtocols != null) {
1095       header.add("Q_ANTIBODY");
1096       header.add("Q_MHC_CLASS");

1100     for (String col : header)
1101       headerLine += "\t" + col;
1102

1104       String label = f.getLabel();
1105       switch (f.getType()) {
1106         case Factor:

1111           break;
1112         default:
1113           break;
1114       }
1115     }

1117       String code = s.getCode();
1118       if (isEntity(code) || SampleCodeFunctions.isQbicBarcode(code)
1119           || SampleCodeFunctions.isMeasurementOfBarcode(code, s.getValueMap().get("SAMPLE TYPE"))) {
1120         Map<String, String> data = s.getValueMap();
1121         String row = s.getCode();
1122         List<String> factors = s.getFactorStringsWithoutLabel();
1123         for (String col : header) {
1124           String val = "";
1125           if (data.containsKey(col))
1126             val = data.get(col);
1127           if (val == null)
1128             val = "";
1129           row += "\t" + val;
1130         }
1131         for (int i = 0; i < factors.size(); i++)
1132           row += "\t" + factors.get(i);
1133         for (int i = factors.size(); i < factorRowSize; i++) {
1134           row += "\t";
1135         }
1136         rows.add(row);
1137       } else {
1138         logger.warn(
1139             code + " will be ignored, it is not a valid QBiC barcode registerable by the wizard.");
1140       }
1141     }
1142     result += headerLine + "\n";
1143     for (String line : rows) {
1144       result += line + "\n";
1145     }
1146     this.tsvContent = result;
1147     return result;
1148   }
