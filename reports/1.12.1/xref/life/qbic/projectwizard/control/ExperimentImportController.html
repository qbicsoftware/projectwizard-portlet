485       private Map<String, Map<String, Object>> fixExtractionProperties(
486           Map<String, Map<String, Object>> extrPrepProperties) {
487         if (extrPrepProperties == null) {
488           return null;
489         }
490         Map<String, Map<String, Object>> res = new HashMap<>();
491         for (String exp : extrPrepProperties.keySet()) {
492           Map<String, Object> props = extrPrepProperties.get(exp);
493           String lysisMethods = "";
494           if (props.containsKey("Q_CELL_LYSIS_METHOD")) {
495             List<String> methods = (List<String>) props.get("Q_CELL_LYSIS_METHOD");
496             lysisMethods = String.join(", ", methods);
497           }
498           if (!lysisMethods.isEmpty()) {
499             props.put("Q_CELL_LYSIS_METHOD", lysisMethods);
500           }
501           res.put(exp, props);
502         }
503         return res;
504       }

506       private Map<String, Map<String, Object>> fixSamplePrepProperties(
507           Map<String, Map<String, Object>> samplePrepProperties) {
508         if (samplePrepProperties == null) {
509           return null;
510         }
511         Map<String, Map<String, Object>> res = new HashMap<>();
512         for (String exp : samplePrepProperties.keySet()) {
513           Map<String, Object> props = samplePrepProperties.get(exp);
514           String digestionInfo = "";
515           if (props.containsKey("Q_DIGESTION_METHOD")) {
516             digestionInfo = (String) props.get("Q_DIGESTION_METHOD");
517             // TODO translate to humanly readable version
518           }
519           if (props.containsKey("Q_DIGESTION_ENZYMES")) {
520             List<String> enzymes = (List<String>) props.get("Q_DIGESTION_ENZYMES");
521             String enzymesString = String.join(", ", enzymes);
522             if (!digestionInfo.isEmpty()) {
523               digestionInfo += ": ";
524             }
525             digestionInfo += enzymesString;
526           }
527           if (!digestionInfo.isEmpty()) {
528             props.put("Q_DIGESTION_METHOD", digestionInfo);
529           }
530           props.remove("Q_DIGESTION_ENZYMES");
531           res.put(exp, props);
532         }
533         return res;
534       }

550   private String replaceChangedMetadata(String tsvContent,
551       Map<String, String> metadataReplacements) {
552     logger.warn("Debug: String replacement");
553     String res = tsvContent;
554     for (String userInput : metadataReplacements.keySet()) {
555       String selectedVocabValue = metadataReplacements.get(userInput);
556 
557       // special characters have to be escaped for replaceAll to work
558       String cleanedInput = Pattern.quote(userInput);
559 
560       logger.warn("raw: " + userInput + ", cleaned input: " + cleanedInput);
561 
562       String in = "(\\t|\\+)" + cleanedInput + "(\\t|\\+)";
563       res = res.replaceAll(in, "$1" + selectedVocabValue + "$2");
564     }
565     return res;

595         prepDesignXML(prep.getTechnologyTypes());
596         break;

598       case Standard:

601         if (!nodes.getFactorsToSamples().keySet().isEmpty())

816   protected String addBarcodesToTSV(List<String> tsv, List<List<ISampleBean>> levels,

818     Set<String> barcodeColumnNames = new HashSet<>(Arrays.asList("QBiC Code", "QBiC Barcode"));

820     String fileNameHeader = null;

824         fileNameHeader = "File Name";
825         break;
826       case Metabolomics_LCMS:

831         break;
832       default:
833         break;
834     }
835

841         if (anltIDPos < 0) {

845         } else {

848             extID = splt[extIDPos];

861           }

864       int filePos = -1;
865       boolean colExists = false;
866       for (String line : tsv) {
867         String[] splt = line.split("\t");
868         // header position
869         if (filePos < 0) {
870           colExists = barcodeColumnNames.contains(splt[0]);
871           filePos = Arrays.asList(splt).indexOf(fileNameHeader);
872           if (colExists) {
873             builder.append(line + "\n");
874           } else {
875             builder.append("QBiC Code\t" + line + "\n");
876           }
877         } else {
878           String file = splt[filePos];
879           String code = fileNameToBarcode.get(file);
880           if (colExists) {
881             builder.append(code + line + "\n");
882           } else {
883             builder.append(code + "\t" + line + "\n");
884           }
885         }
886       }
887     }
888     return builder.toString();

1541   private String createUniqueIDFromSampleMetadata(ISampleBean b) {
1542     String id = null;
1543     if (b.getMetadata().get("Q_EXTERNALDB_ID") != null) {
1544       id = (String) b.getMetadata().get("Q_EXTERNALDB_ID");
1545     }
1546     String secID = b.getSecondaryName();
1547     if (getImportType().equals(ExperimentalDesignType.Proteomics_MassSpectrometry)) {
1548       // MS import format may contain same name for protein and digested peptide samples, both at
1549       // the base, pooled and fraction/cycle level
1550       // thus, we add the sample type to the external db id in order to create unique sample ids:
1551       if (b.getType().equals(SampleType.Q_TEST_SAMPLE)) {
1552         id = id + secID + b.getMetadata().get("Q_SAMPLE_TYPE");
1553         for (String parentCode : b.getParentIDs()) {
1554           if (importCodeToSampleBean.containsKey(parentCode)) {
1555             id = createUniqueIDFromSampleMetadata(importCodeToSampleBean.get(parentCode)) + id;
1556           } else {
1557             // TODO use sample object from barcode?
1558             id = parentCode + id;
1559           }
1560           break;
1561         }
1562       }
1563       if (b.getType().equals(SampleType.Q_MS_RUN)) {
1564         id = b.getType() + b.getCode();
1565       }
1566     }
1567     if (id == null) {
1568       return b.getCode();
1569     }
1570     return id;
