34  public class UploadComponent extends VerticalLayout
35      implements Upload.SucceededListener, Upload.FailedListener, Upload.Receiver,
36      Upload.ProgressListener, Upload.FinishedListener, StartedListener {

38    /**
39     * 
40     */
41    private static final long serialVersionUID = -3228100993055350771L;
42    protected Upload upload;
43    protected String directory;
44    protected String user;
45    protected File file;
46    protected long maxSize; // In bytes. 100Kb = 100000
47    protected ProgressBar progressIndicator; // May be null

49    protected Long contentLength;
50    protected Button cancelProcessing;
51    protected HorizontalLayout processingLayout;
52  
53    private static final Logger logger = LogManager.getLogger(UploadComponent.class);

55  
56    public UploadComponent(String fieldCaption, String buttonCaption, String directoryParam,
57        String user, int maxSize) {
58      upload = new Upload(fieldCaption, null);
59      this.user = user;
60      this.addComponent(upload);
61      this.maxSize = maxSize;
62      upload.setReceiver(this);
63      this.directory = directoryParam;
64      // this.targetPrefix = targetPrefix;
65      upload.setButtonCaption(buttonCaption);
66      upload.addSucceededListener(this);
67      upload.addFailedListener(this);
68      upload.addProgressListener(this);
69      upload.addFinishedListener(this);
70      upload.addStartedListener(this);
71  
72      processingLayout = new HorizontalLayout();
73      processingLayout.setSpacing(true);
74      processingLayout.setVisible(false);
75      this.addComponent(processingLayout);
76  
77      progressIndicator = new ProgressBar();
78      progressIndicator.setWidth("100%");
79      processingLayout.addComponent(progressIndicator);
80  
81      cancelProcessing = new Button("cancel", new Button.ClickListener() {
82        /**
83         * 
84         */
85        private static final long serialVersionUID = -6812538024652192780L;
86  
87        @Override
88        public void buttonClick(ClickEvent event) {
89          cancelled = true;
90          upload.interruptUpload();
91        }
92      });
93  
94      cancelProcessing.setStyleName("small");
95      processingLayout.addComponent(cancelProcessing);
96    }

98    @Override
99    public OutputStream receiveUpload(String filename, String MIMEType) {
100     filename = FilenameUtils.getName(filename);
101     if (filename.isEmpty()) {
102       upload.interruptUpload();
103     }
104     FileOutputStream fos = null;
105     Date date = new java.util.Date();
106     String timeStamp = new SimpleDateFormat("HHmmssS").format(new Timestamp(date.getTime()));
107     file = new File(directory, user + "_" + timeStamp + "_" + filename);
108     try {
109       fos = new FileOutputStream(file);
110     } catch (final java.io.FileNotFoundException e) {
111       throw new RuntimeException(e);
112     }
113     return fos; // Return the output stream to write to
114   }
115 
116   public Upload getUploadComponent() {
117     return upload;
118   }
119 
120   @Override
121   public void updateProgress(long readBytes, long contentLength) {
122     this.contentLength = contentLength;
123     if (maxSize < contentLength) {
124       upload.interruptUpload();
125       return;
126     }
127 
128     processingLayout.setVisible(true);
129     progressIndicator.setValue(new Float(readBytes / (float) contentLength));
130   }
131 
132   @Override
133   public void uploadFinished(FinishedEvent event) {
134     processingLayout.setVisible(false);
135   }
136 
137 
138   @Override
139   public void uploadFailed(FailedEvent event) {
140     processingLayout.setVisible(false);
141     if (event.getFilename().isEmpty()) {
142       Styles.notification("No file selected", "Please select a file before adding it.",
143           NotificationType.ERROR);
144 
145       logger.info("Upload was cancelled due to no file selected.");
146     } else if (contentLength != null && maxSize < contentLength) {
147       Styles.notification("File too large", "Your file is " + contentLength / 1000
148           + "Kb long. Maximum file size is " + maxSize / 1000 + "Kb", NotificationType.ERROR);
149 
150       logger.info("Upload was cancelled due to file exceeding size limit.");
151     } else if (cancelled) {
152       // Nothing to do...
153     } else {
154       StringWriter sw = new StringWriter();
155       PrintWriter pw = new PrintWriter(sw);
156       if (event.getReason() != null)
157         event.getReason().printStackTrace(pw);
158       logger.error("Upload cancelled due to error.");
159       Styles.notification("There was a problem uploading your file.", "", NotificationType.ERROR);
160     }
161 
162     try {
163       file.delete();
164     } catch (Exception e) {
165       // Silent exception. If we can't delete the file, it's not big problem. Maybe the file did not
166       // even exist.
167     }
168   }
