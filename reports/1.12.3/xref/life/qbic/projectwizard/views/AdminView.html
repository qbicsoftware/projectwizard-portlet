123   private void initButtons() {
124     createSpace.addClickListener(new Button.ClickListener() {
125 
126       /**
127        * 
128        */
129       private static final long serialVersionUID = -6870391592753359641L;
130 
131       @Override
132       public void buttonClick(ClickEvent event) {
133         space.validate();
134         if (space.isValid()) {
135           createSpace.setEnabled(false);
136           String space = getSpace().toUpperCase();
137           if (!openbis.spaceExists(space)) {
138             HashMap<OpenbisSpaceUserRole, ArrayList<String>> roleInfos =
139                 new HashMap<OpenbisSpaceUserRole, ArrayList<String>>();
140             if (getUsers().size() > 0)
141               roleInfos.put(OpenbisSpaceUserRole.USER, getUsers());
142             registrator.registerSpace(space, "test description", roleInfos);//TODO
143             // wait few seconds, then check for a maximum of timeout seconds, if space was created
144             int timeout = 5;
145             int wait = 2;
146             try {
147               Thread.sleep(wait * 1000);
148             } catch (InterruptedException e) {
149               // TODO Auto-generated catch block
150               e.printStackTrace();
151             }
152             while (timeout > 0 && !openbis.spaceExists(space)) {
153               timeout -= 1;
154               try {
155                 Thread.sleep(1000);
156               } catch (InterruptedException e) {
157                 // TODO Auto-generated catch block
158                 e.printStackTrace();
159               }
160             }
161             if (openbis.spaceExists(space)) {
162               Styles.notification("Space created", "The space " + space + " has been created!",
163                   NotificationType.SUCCESS);
164               handleSpaceCreationSuccess();
165             } else {
166               Styles.notification("Problem creating space",
167                   "There seems to have been a problem while creating the space. Do the specified users already exist in openbis? If not, create them.",
168                   NotificationType.ERROR);
169             }
170             createSpace.setEnabled(true);
171           }
172         }
173       }
174     });
175   }
176 
177   protected void handleSpaceCreationSuccess() {
178     users.setValue("");
179     space.setValue("");
180     mainController.resetSpaces();
181   }

189       return new ArrayList<String>(Arrays.asList(users.getValue().split("\n")));
190     else
191       return new ArrayList<String>();
192   }
