18  package life.qbic.projectwizard.io;

20  
21  import java.io.File;
22  import java.io.FileNotFoundException;
23  import java.util.ArrayList;
24  import java.util.List;
25  import java.util.Scanner;
26  
27  import life.qbic.projectwizard.uicomponents.LabelingMethod;
28  
29  public class QuantificationMethodVocabularyParser {
30  
31    public List<LabelingMethod> parseQuantificationMethods(File file) {
32      Scanner scanner = null;
33      try {
34        scanner = new Scanner(file);
35      } catch (FileNotFoundException e) {
36        // TODO Auto-generated catch block
37        e.printStackTrace();
38      }
39      List<LabelingMethod> methods = new ArrayList<LabelingMethod>();
40      List<String> reagents = new ArrayList<String>();
41      String name = "";
42      while (scanner.hasNext()) {
43        String line = scanner.nextLine();
44        if (!line.startsWith("\t")) {
45          if (!name.isEmpty()) {
46            methods.add(new LabelingMethod(name, reagents));
47            reagents = new ArrayList<String>();
48          }
49          name = line.trim();
50        } else {
51          reagents.add(line.trim());
52        }
53      }
54      methods.add(new LabelingMethod(name, reagents));
55      scanner.close();
56      return methods;
57    }
58  
59  }
