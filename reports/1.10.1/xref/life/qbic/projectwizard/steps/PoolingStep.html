43  public class PoolingStep implements WizardStep {
44  
45    private VerticalLayout main;
46    private Steps type;
47    private DragDropPoolComponent pooling;
48    private List<DragDropPoolComponent> poolings;
49    private TabSheet instances;
50    private MSExperimentModel model;
51    private static final Logger logger = LogManager.getLogger(PoolingStep.class);
52  
53    public PoolingStep(Steps poolStep) {
54      instances = new TabSheet();
55      instances.setStyleName(ValoTheme.TABSHEET_FRAMED);
56      poolings = new ArrayList<DragDropPoolComponent>();
57      this.type = poolStep;
58      main = new VerticalLayout();
59      main.setSpacing(true);
60      main.setMargin(true);
61      pooling = new DragDropPoolComponent(getPoolPrefix(poolStep));
62      main.addComponent(instances);
63    }
64  
65    private String getPoolPrefix(Steps poolStep) {
66      String name;
67      switch (type) {
68        case Extract_Pooling:
69          name = "Extr-";
70          break;
71        case Test_Sample_Pooling:
72          name = "Prep-";
73          break;
74        case Protein_Fractionation_Pooling:
75          name = "Fraction-";
76          break;
77        case Peptide_Fractionation_Pooling:
78          name = "Fraction-";
79          break;
80        default:
81          name = "";
82          break;
83      }
84      return name;
85    }
86  
87    @Override
88    public String getCaption() {
89      String name;
90      switch (type) {
91        case Extract_Pooling:
92          name = "Extr. Pooling";
93          break;
94        case Test_Sample_Pooling:
95          name = "Prep. Pooling";
96          break;
97        case Protein_Fractionation_Pooling:
98          name = "Fract. Pooling";
99          break;
100       case Peptide_Fractionation_Pooling:
101         name = "Fract. Pooling";
102         break;
103       default:
104         name = "";
105         break;
106     }
107     return name;
108   }
109 
110   @Override
111   public Component getContent() {
112     return main;
113   }
114 
115   @Override
116   public boolean onAdvance() {
117     if (pooling.getPools().isEmpty()) {
118       Styles.notification("No pooled samples",
119           "Please create at least one pool containing samples or uncheck pooling in the previous step(s).",
120           NotificationType.ERROR);
121       return false;
122     } else
123       return true;
124   }
125 
126   @Override
127   public boolean onBack() {
128     resetStep();
129     return true;
130   }
131 
132   public void setSamples(List<List<AOpenbisSample>> sampleGroups, Steps poolingType) {
133     for (List<AOpenbisSample> group : sampleGroups) {
134       String type = group.get(0).getValueMap().get("Q_SAMPLE_TYPE");
135       pooling = new DragDropPoolComponent(getPoolPrefix(poolingType));
136       pooling.initConditionsAndSetSamples(group);
137       poolings.add(pooling);
138       instances.addTab(pooling, type);
139     }
140   }
141 
142   public Map<String, List<AOpenbisSample>> getPools() {
143     Map<String, List<AOpenbisSample>> res = new HashMap<String, List<AOpenbisSample>>();
144     for (DragDropPoolComponent pooling : poolings) {
145       res.putAll(pooling.getPools());
146     }
147     return res;
148   }
149 
150   public void resetStep() {
151     poolings = new ArrayList<DragDropPoolComponent>();
152     main.removeComponent(instances);
153     instances = new TabSheet();
154     main.addComponent(instances);
155   }
156 
157   // sets Samples from the last analyte level (last step) of the current ms experiment life.qbic.projectwizard.model to be
158   // used in the pooling.
159   // in the future pooling between samples created in different steps might be considered
160   public void setPreliminaryExperiments(MSExperimentModel msExperimentModel, Steps poolingType) {
161     this.model = msExperimentModel;
162     List<List<AOpenbisSample>> allSamples = new ArrayList<List<AOpenbisSample>>();
163     List<AOpenbisSample> currentSamples = new ArrayList<AOpenbisSample>();
164     for (ExperimentModel fract : msExperimentModel.getLastStepAnalytes()) {
165       currentSamples.addAll(fract.getSamples());
166     }
167     allSamples.add(currentSamples);
168     setSamples(allSamples, poolingType);
169   }
170 
171   public MSExperimentModel getPreliminarySamples() {
172     String analyte = "unknown";
173     if (type.equals(Steps.Protein_Fractionation_Pooling))
174       analyte = "PROTEINS";
175     if (type.equals(Steps.Peptide_Fractionation_Pooling))
176       analyte = "PEPTIDES";
177     List<AOpenbisSample> samples = new ArrayList<AOpenbisSample>();
178     Map<String, List<AOpenbisSample>> pools = getPools();
179     for (String name : pools.keySet()) {
180       samples.add(new OpenbisTestSample(1, pools.get(name), analyte, name, "",
181           new ArrayList<Property>(), ""));
182     }
183     ExperimentModel exp = new ExperimentModel(1, samples);
184     List<ExperimentModel> exps = new ArrayList<ExperimentModel>(Arrays.asList(exp));
185     this.model.addAnalyteStepExperiments(exps);
186     return model;
187   }
