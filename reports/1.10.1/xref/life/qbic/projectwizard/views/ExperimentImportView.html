210   private Component createTSVDownloadComponent(ExperimentalDesignType type) {
211     if (type.equals(ExperimentalDesignType.ISA)) {
212       VerticalLayout v = new VerticalLayout();
213       Label l = new Label("For the ISA specification see:");
214       Link link = new Link("http://isa-specs.readthedocs.io/en/latest/isatab.html",
215           new ExternalResource("http://isa-specs.readthedocs.io/en/latest/isatab.html"));
216       // Open the URL in a new window/tab
217       link.setTargetName("_blank");
218       v.addComponent(l);
219       v.addComponent(link);
220       return v;
221     } else {
222       VerticalLayout v = new VerticalLayout();
223       v.setSpacing(true);
224       Label l = new Label(type.getDescription());
225       l.setWidth("300px");
226       v.addComponent(l);
227       Button button = new Button("Download Example");
228       v.addComponent(button);
229 
230       final File example = new File(
231           getClass().getClassLoader().getResource("examples/" + type.getFileName()).getFile());
232       FileDownloader tsvDL = new FileDownloader(new FileResource(example));
233       tsvDL.extend(button);
234       return v;
235     }
236   }

261   public void registrationDone(boolean sqlDown, String errors) {
262     // TODO when adding mysql metadata handle sql down
263     if (errors.isEmpty()) {
264       logger.info("Sample registration complete!");
265       Styles.notification("Registration complete!", "Registration of samples complete.",
266           NotificationType.SUCCESS);
267       register.setEnabled(false);
268       switch (getSelectedDesignOption()) {
269         case Standard:
270         case Proteomics_MassSpectrometry:
271         case MHC_Ligands_Finished:
272           downloadTSV.setEnabled(true);
273           break;
274         default:
275           break;
276       }
277     } else {
278       String feedback = "Sample registration could not be completed. Reason: " + errors;
279       logger.error(feedback);
280       Styles.notification("Registration failed!", feedback, NotificationType.ERROR);
281     }
282   }

305   public ExperimentalDesignType getSelectedDesignOption() {
306     if (importOptions.getValue() != null) {
307       switch (importOptions.getValue().toString()) {
308         case "QBiC":
309           return ExperimentalDesignType.QBIC;
310         case "Standard":
311           return ExperimentalDesignType.Standard;
312         case "MHC Ligandomics (preparation)":
313           return ExperimentalDesignType.MHC_Ligands_Plan;
314         case "MHC Ligandomics":
315           return ExperimentalDesignType.MHC_Ligands_Finished;
316         case "Mass Spec Proteomics":
317           return ExperimentalDesignType.Proteomics_MassSpectrometry;
318         case "ISA-Tab (prototype)":
319           return ExperimentalDesignType.ISA;
320         default:
321           return ExperimentalDesignType.Standard;
322       }
323     } else
324       return null;
325   }
