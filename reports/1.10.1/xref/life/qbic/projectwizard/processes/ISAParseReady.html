16  package life.qbic.projectwizard.processes;
17  
18  import java.util.List;
19  
20  import org.isatools.isacreator.model.Study;
21  
22  import life.qbic.projectwizard.control.ExperimentImportController;
23  
24  public class ISAParseReady implements Runnable {
25  
26    private ExperimentImportController importer;
27    private List<Study> studies;
28    private String error;
29  
30    public ISAParseReady(ExperimentImportController importer, List<Study> studies, String error) {
31      this.importer = importer;
32      this.studies = studies;
33      this.error = error;
34    }
35  
36    @Override
37    public void run() {
38      importer.isaPrepComplete(studies, error);
39    }
40  
41  }
