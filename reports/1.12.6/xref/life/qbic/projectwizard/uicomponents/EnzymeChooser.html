18  package life.qbic.projectwizard.uicomponents;
19  
20  import java.util.List;
21  
22  import com.vaadin.shared.ui.combobox.FilteringMode;
23  import com.vaadin.ui.ComboBox;
24  import com.vaadin.ui.VerticalLayout;
25  
26  import life.qbic.portal.Styles;
27  
28  public class EnzymeChooser extends VerticalLayout {
29  
30    /**
31     * 
32     */
33    private static final long serialVersionUID = 284773693595639437L;
34    /**
35     * 
36     */
37    private ComboBox chooser;
38  
39    public EnzymeChooser(List<String> options) {
40      chooser = new ComboBox();
41      chooser.addItems(options);
42      chooser.setFilteringMode(FilteringMode.CONTAINS);
43  //    chooser.setNullSelectionAllowed(false);
44      chooser.setStyleName(Styles.boxTheme);
45      addComponent(chooser);
46      setSpacing(true);
47    }
48  
49    public boolean chooserSet() {
50      return chooser.getValue() != null;
51    }
52  
53    public boolean isSet() {
54      return chooser.getItemIds().contains(chooser.getValue());
55    }
56  
57    public String getEnzyme() {
58      if (chooser.getValue() != null)
59        return chooser.getValue().toString();
60      else
61        return null;
62    }
63  
64    public void reset() {
65      chooser.setValue(chooser.getNullSelectionItemId());
66    }
67  }
