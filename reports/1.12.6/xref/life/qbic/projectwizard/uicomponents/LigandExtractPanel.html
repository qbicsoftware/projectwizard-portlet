96    public Map<String, Map<String, Object>> getExperimentalProperties() {
97      Map<String, Map<String, Object>> res = new HashMap<String, Map<String, Object>>();
98      Map<String, MHCLigandExtractionProtocol> infos = getAntibodyInfos();
99      for (String key : infos.keySet())
100       res.put(key, translateToProperties(infos.get(key)));
101     return res;
102   }

104   private Map<String, Object> translateToProperties(MHCLigandExtractionProtocol prot) {
105     Map<String, Object> res = new HashMap<String, Object>();
106 
107     try {
108       res.put("Q_SAMPLE_MASS", Double.parseDouble(prot.getInputSampleMass()));
109     } catch (Exception e) {
110       logger.warn("Sample mass not set or not parsable, ignoring.");
111     }
112 
113     try {
114       SimpleDateFormat dateformat = new SimpleDateFormat("dd-MM-yyyy");
115       Date date = prot.getPrepDate();
116       if (date != null) {
117         String dateString = dateformat.format(date);
118         res.put("Q_PREPARATION_DATE", dateString);
119       }
120     } catch (IllegalArgumentException e) {
121       logger.warn("No valid preparation date input. Not setting Date for this experiment.");
122     }
123 
124     List<String> antibodies = prot.getAntibodies();
125     for (int i = 1; i <= antibodies.size(); i++) {
126       String antibody = antibodies.get(i - 1);
127       res.put("Q_MHC_ANTIBODY_COL" + Integer.toString(i), antibody);
128       try {
129         double mass = Double.parseDouble(prot.getAntiBodyMasses().get(i - 1));
130         res.put("Q_MHC_ANTIBODY_MASS_COL" + Integer.toString(i), mass);
131       } catch (Exception e) {
132         logger.warn("Antibody mass not set or not parsable, ignoring.");
133       }
134 
135     }
136     return res;
137   }

155   public void setTissueSamples(List<AOpenbisSample> extracts) {
156     extractionExperiments.removeAllItems();
157     tableIdToBarcode = new HashMap<Integer, String>();
158     int i = 0;
159     for (AOpenbisSample s : extracts) {
160       i++;
161       tableIdToBarcode.put(i, s.getCode());
162 
163       List<Object> row = new ArrayList<Object>();
164 
165       row.add(s.getQ_SECONDARY_NAME());
166       row.add(generateTableIntegerInput());
167       DateField date = new DateField();
168       date.setDateFormat("dd.MM.yy");
169       date.setWidth("110px");
170       date.setStyleName(Styles.fieldTheme);
171       row.add(date);
172       row.add(generateTableAntibodyBox());
173       row.add(generateTableIntegerInput());
174       row.add(generateTableAntibodyBox());
175       row.add(generateTableIntegerInput());
176       row.add(generateTableAntibodyBox());
177       row.add(generateTableIntegerInput());
178 
179       extractionExperiments.addItem(row.toArray(new Object[row.size()]), i);
180     }
181     extractionExperiments.setPageLength(extracts.size());
182   }

184   public Map<String, MHCLigandExtractionProtocol> getAntibodyInfos() {
185     Map<String, MHCLigandExtractionProtocol> res =
186         new HashMap<String, MHCLigandExtractionProtocol>();
187 
188     for (Iterator i = extractionExperiments.getItemIds().iterator(); i.hasNext();) {
189       // Get the current item identifier, which is an integer.
190       int iid = (Integer) i.next();
191 
192       // Now get the actual item from the table.
193       Item item = extractionExperiments.getItem(iid);
194 
195       String source = tableIdToBarcode.get(iid);// came from this sample
196       TextField mass = (TextField) item.getItemProperty("Mass [mg]").getValue();
197       String inputSampleMass = mass.getValue();
198       DateField d = (DateField) item.getItemProperty("Date").getValue();
199       Date date = d.getValue();
200       ComboBox ab1 = (ComboBox) item.getItemProperty("Antibody 1").getValue();
201       TextField mass1 = (TextField) item.getItemProperty("Mass 1 [mg]").getValue();
202       ComboBox ab2 = (ComboBox) item.getItemProperty("Antibody 2").getValue();
203       TextField mass2 = (TextField) item.getItemProperty("Mass 2 [mg]").getValue();
204       ComboBox ab3 = (ComboBox) item.getItemProperty("Antibody 3").getValue();
205       TextField mass3 = (TextField) item.getItemProperty("Mass 3 [mg]").getValue();
206       List<String> antibodies = new ArrayList<String>();
207       List<String> antibodyMasses = new ArrayList<String>();
208 
209       if (ab1.getValue() != null) {
210         antibodies.add(ab1.getValue().toString());
211         antibodyMasses.add(mass1.getValue());
212       }
213       if (ab2.getValue() != null) {
214         antibodies.add(ab2.getValue().toString());
215         antibodyMasses.add(mass2.getValue());
216       }
217       if (ab3.getValue() != null) {
218         antibodies.add(ab3.getValue().toString());
219         antibodyMasses.add(mass3.getValue());
220       }
221 
222       res.put(source, new MHCLigandExtractionProtocol(inputSampleMass, date, antibodies,
223           antibodyMasses));
224     }
225     return res;
226   }

228   public boolean isValid() {
229     boolean res = true;
230     String error = "";
231     for (Iterator i = extractionExperiments.getItemIds().iterator(); i.hasNext();) {
232       // Get the current item identifier, which is an integer.
233       int iid = (Integer) i.next();
234 
235       // Now get the actual item from the table.
236       Item item = extractionExperiments.getItem(iid);
237 
238       TextField mass = (TextField) item.getItemProperty("Mass [mg]").getValue();
239       try {
240         Integer.parseInt(mass.getValue());
241       } catch (NumberFormatException e) {
242         res = false;
243         error = "Sample mass has to be a number!";
244       }
245       DateField d = (DateField) item.getItemProperty("Date").getValue();
246 
247       if (d.getValue() == null) {
248         error = "Please select preparation dates for all samples!";
249       }
250       ComboBox ab1 = (ComboBox) item.getItemProperty("Antibody 1").getValue();
251       TextField mass1 = (TextField) item.getItemProperty("Mass 1 [mg]").getValue();
252       ComboBox ab2 = (ComboBox) item.getItemProperty("Antibody 2").getValue();
253       TextField mass2 = (TextField) item.getItemProperty("Mass 2 [mg]").getValue();
254       ComboBox ab3 = (ComboBox) item.getItemProperty("Antibody 3").getValue();
255       TextField mass3 = (TextField) item.getItemProperty("Mass 3 [mg]").getValue();
256 
257       String antibodyError = "Please choose at least one antibody and fill in the mass.";
258 
259       if (ab1.getValue() != null && !mass1.getValue().isEmpty()) {
260         try {
261           Integer.parseInt(mass.getValue());
262         } catch (NumberFormatException e) {
263           res = false;
264           error = "Antibody 1 mass has to be a number!";
265         }
266       } else {
267         res = false;
268         error = antibodyError;
269       }
270       if (ab2.getValue() != null && !mass2.getValue().isEmpty()) {
271         try {
272           Integer.parseInt(mass.getValue());
273         } catch (NumberFormatException e) {
274           res = false;
275           error = "Antibody 2 mass has to be a number!";
276         }
277       }
278       if (ab3.getValue() != null && !mass3.getValue().isEmpty()) {
279         try {
280           Integer.parseInt(mass.getValue());
281         } catch (NumberFormatException e) {
282           res = false;
283           error = "Antibody 3 mass has to be a number!";
284         }
285       }
286     }
287 
288     if (!res) {
289       Notification n = new Notification(error);
290       n.setStyleName(ValoTheme.NOTIFICATION_CLOSABLE);
291       n.setDelayMsec(-1);
292       n.show(UI.getCurrent().getPage());
293       return false;
294     } else
295       return true;
296   }
