26    public RegisterableProject(List<List<ISampleBean>> tsvSampleHierarchy, String description,
27        List<OpenbisExperiment> informativeExperiments, boolean isPilot) {
28      this.description = description;
29      this.experiments = new ArrayList<RegisterableExperiment>();
30      Map<String, OpenbisExperiment> knownExperiments = new HashMap<String, OpenbisExperiment>();
31      for (OpenbisExperiment e : informativeExperiments) {
32        knownExperiments.put(e.getExperimentCode(), e);
33      }
34      Map<String, RegisterableExperiment> expMap = new HashMap<String, RegisterableExperiment>();
35  
36      for (List<ISampleBean> inner : tsvSampleHierarchy) {
37        // needed since we collect some samples that don't have the same experiment now - TODO not the
38        // best place here
39        if (!inner.isEmpty()) {
40          ISampleBean sa = inner.get(0);
41          this.space = sa.getSpace();
42          this.code = sa.getProject();
43          for (ISampleBean s : inner) {
44            String expCode = s.getExperiment();
45            // we know this experiment, add the current sample
46            if (expMap.containsKey(expCode)) {
47              RegisterableExperiment exp = expMap.get(expCode);
48              exp.addSample(s);
49            } else {
50              ExperimentType expType = null;
51              Map<String, Object> metadata = new HashMap<String, Object>();
52              if (knownExperiments.containsKey(expCode)) {
53                OpenbisExperiment exp = knownExperiments.get(expCode);
54                expType = exp.getType();
55                metadata = exp.getMetadata();
56              } else {
57                // experiment is new, get the metadata, create it and put it into the map
58                expType = SampleCodeFunctions.sampleTypesToExpTypes.get(s.getType());
59              }
60  
61              if (!notPilotable.contains(expType)) {
62                metadata.put("Q_IS_PILOT", isPilot);
63              }
64              RegisterableExperimentgisterableExperiment.html#RegisterableExperiment">RegisterableExperiment e = new RegisterableExperiment(s.getExperiment(), expType,
65                  new ArrayList<ISampleBean>(Arrays.asList(s)), metadata);
66              this.experiments.add(e);
67              expMap.put(expCode, e);
68            }
69          }
70        }
71      }
72    }
