50    private static String getScientificNameFromXML(InputStream stream)
51        throws XPathExpressionException, SAXException, IOException, ParserConfigurationException {
52      DocumentBuilderFactory builderFactory = DocumentBuilderFactory.newInstance();
53      DocumentBuilder builder = builderFactory.newDocumentBuilder();
54      Document xmlDocument = builder.parse(stream);
55      XPath xPath = XPathFactory.newInstance().newXPath();
56  
57      String expression = "/TaxaSet/Taxon/ScientificName";
58      NodeList nodeList =
59          (NodeList) xPath.compile(expression).evaluate(xmlDocument, XPathConstants.NODESET);
60  
61      String name = null;
62      for (int i = 0; i < nodeList.getLength(); i++) {
63        name = nodeList.item(i).getFirstChild().getNodeValue();
64      }
65      return name;
66    }

68    public List<NCBITerm> getSpecies(String search)
69        throws IOException, XPathExpressionException, SAXException, ParserConfigurationException {
70      List<String> candidates = new ArrayList<String>();
71      candidates.add(search);
72      try {
73        Integer.parseInt(search);
74      } catch (NumberFormatException e) {
75        candidates = getTaxIDsByName(search);
76      }
77      List<NCBITerm> res = new ArrayList<NCBITerm>();
78      for (String id : candidates)
79        res.add(searchByID(id));
80  //    Collections.sort(res);
81      return res;
82    }

84    private List<String> getTaxIDsByName(String search)
85        throws IOException, XPathExpressionException, SAXException, ParserConfigurationException {
86      search = URLEncoder.encode(search, "UTF-8").replace("+", "%20");
87  
88      InputStream stream =
89          httpCall(ncbiSearchRestURL.concat(search).concat("[subtree]").concat("&retmode=xml"));
90  
91      List<String> taxIDs = searchTaxIDsInXML(stream);
92  
93      return taxIDs;
94    }

114   private NCBITerm searchByID(String id)
115       throws IOException, XPathExpressionException, SAXException, ParserConfigurationException {
116     InputStream stream = httpCall(ncbiTaxRestURL.concat(id).concat("&retmode=xml"));
117 
118     String name = getScientificNameFromXML(stream);
119 
120     return new NCBITerm(id, name, "");
121   }

123   public static InputStream httpCall(String url) throws IOException {
124     final HttpURLConnection connection = (HttpURLConnection) (new URL(url).openConnection());
125     connection.setRequestMethod("GET");
126     connection.connect();
127     return connection.getInputStream();
128   }

130   public String getKingdomTerm(String taxID)
131       throws IOException, ParserConfigurationException, SAXException, XPathExpressionException {
132     InputStream stream =
133         httpCall("https://eutils.ncbi.nlm.nih.gov/entrez/eutils/efetch.fcgi?db=taxonomy&id="
134             .concat(taxID).concat("&retmode=xml"));
135     DocumentBuilderFactory builderFactory = DocumentBuilderFactory.newInstance();
136     DocumentBuilder builder = builderFactory.newDocumentBuilder();
137     Document xmlDocument = builder.parse(stream);
138     XPath xPath = XPathFactory.newInstance().newXPath();
139 
140     String expression = "/TaxaSet/Taxon/Lineage";
141     NodeList nodeList =
142         (NodeList) xPath.compile(expression).evaluate(xmlDocument, XPathConstants.NODESET);
143 
144     String lineage = nodeList.item(0).getFirstChild().getNodeValue();
145     // TODO use xpath
146     for (String key : kingdomToCategory.keySet()) {
147       if (lineage.contains(key))
148         return kingdomToCategory.get(key).toString();
149     }
150     return "unknown";
151   }
