91    protected void initTable() {
92      List<Experiment> infoExps =
93          openbis.getExperimentsOfType(ExperimentType.Q_PROJECT_DETAILS.name());
94      List<Sample> infoSamps = openbis.getSamplesOfType(SampleType.Q_ATTACHMENT_SAMPLE.name());
95      List<Project> projects = openbis.listProjects();
96  
97      for (Experiment exp : infoExps) {
98        if (exp.getProperties().containsKey("Q_EXPERIMENTAL_SETUP")) {
99          projectInfoExpsWithDesignXML.add(exp.getIdentifier().getIdentifier());
100       }
101     }
102     for (Sample s : infoSamps) {
103       projectInfoSampsWithDesignXML.add(s.getCode());
104     }
105 
106     for (Project project : projects) {
107       String space = project.getSpace().getCode();
108       String code = project.getCode();
109       if (spaceToProjects.containsKey(space)) {
110         spaceToProjects.get(space).add(code);
111       } else {
112         spaceToProjects.put(space, new HashSet<>(Arrays.asList(code)));
113       }
114     }
115 
116     for (String space : spaceToProjects.keySet()) {
117       Set<String> codes = spaceToProjects.get(space);
118       int numProjects = codes.size();
119       int convertedProjects = 0;
120       for (String p : codes) {
121         String id = ExperimentCodeFunctions.getInfoExperimentID(space, p);
122         if (projectInfoExpsWithDesignXML.contains(id)
123             && projectInfoSampsWithDesignXML.contains(p + "000")) {
124           convertedProjects++;
125         }
126       }
127       // int percentConverted = 100 * convertedProjects / numProjects;
128 
129       List<Object> row = new ArrayList<>();
130       row.add(space);
131       row.add(Integer.toString(convertedProjects) + "/" + Integer.toString(numProjects));
132       // row.add(percentConverted);
133       projectTable.addItem(row.toArray(new Object[row.size()]), space);
134     }
135     Object[] properties = {"Space Name"};
136     boolean[] ordering = {true};
137     projectTable.sort(properties, ordering);
138     addComponent(projectTable);
139     addComponent(convert);
140     convert.addClickListener(new ClickListener() {
141 
142       @Override
143       public void buttonClick(ClickEvent event) {
144         if (projectTable.getValue() != null) {
145           conversionPressed();
146           List<String> spaces = new ArrayList<>((Set<String>) projectTable.getValue());
147           List<Set<String>> existingPerSpace = new ArrayList<>();
148           for (String space : spaces) {
149             Set<String> existing = new HashSet<>();
150             for (String project : spaceToProjects.get(space)) {
151               String id = ExperimentCodeFunctions.getInfoExperimentID(space, project);
152               if (projectInfoExpsWithDesignXML.contains(id)
153                   && projectInfoSampsWithDesignXML.contains(project + "000")) {
154                 existing.add(project);
155               }
156             }
157             existingPerSpace.add(existing);
158           }
159           convertInBackground(spaces, existingPerSpace);
160         }
161       }
162     });
163 
164     addComponent(info);
165     addComponent(bar);
166     info.setVisible(false);
167     bar.setVisible(false);
168   }

170   private void convertInBackground(List<String> spaces, List<Set<String>> existing) {
171     ExperimentalDesignConversionView caller = this;
172 
173     Thread t = new Thread(new Runnable() {
174       volatile int current = -1;
175 
176       @Override
177       public void run() {
178         int numProjects = 0;
179         for (int i = 0; i < spaces.size(); i++) {
180           numProjects += spaceToProjects.get(spaces.get(i)).size() - existing.get(i).size();
181         }
182         info.setCaption("Converting " + numProjects + " projects in " + spaces.size() + " spaces.");
183         UI.getCurrent().access(new UpdateProgressBar(bar, info, 0.01));
184 
185         final int todo = numProjects;
186 
187         for (int i = 0; i < spaces.size(); i++) {
188           for (Project pr : openbis.getProjectsOfSpace(spaces.get(i))) {
189             String project = pr.getCode();
190             if (!existing.get(i).contains(project)) {
191               try {
192                 current++;
193                 double frac = current * 1.0 / todo;
194                 UI.getCurrent().access(new UpdateProgressBar(bar, info, frac));
195                 convertMissingXMLsForProject(spaces.get(i), project);
196               } catch (IllegalArgumentException | JAXBException | InterruptedException e) {
197                 // TODO Auto-generated catch block
198                 e.printStackTrace();
199               }
200             }
201           }
202         }
203 
204         UI.getCurrent().setPollInterval(-1);
205         UI.getCurrent().access(new ConversionReadyRunnable(caller, spaces));
206       }
207     });
208     t.start();
209     UI.getCurrent().setPollInterval(100);
210   }

227   private void updateLocalData(List<String> spaces) {
228     for (String space : spaces) {
229       for (String project : spaceToProjects.get(space)) {
230         String id = ExperimentCodeFunctions.getInfoExperimentID(space, project);
231         projectInfoExpsWithDesignXML.add(id);
232       }
233       Item item = projectTable.getItem(space);
234       String numProjects = Integer.toString(spaceToProjects.get(space).size());
235       String allConverted = numProjects + "/" + numProjects;
236       item.getItemProperty("Converted projects").setValue(allConverted);
237     }

240   private void convertMissingXMLsForProject(String space, String project)
241       throws IllegalArgumentException, JAXBException, InterruptedException {
242     Set<String> types = new HashSet<>(Arrays.asList("Q_BIOLOGICAL_SAMPLE", "Q_BIOLOGICAL_ENTITY",
243         "Q_TEST_SAMPLE", "Q_MHC_LIGAND_EXTRACT"));
244     Map<String, Map<Pair<String, String>, List<String>>> expDesign = new HashMap<>();
245     Map<String, List<Qproperty>> otherProps = new HashMap<>();
246     String TARGET_EXPERIMENT = ExperimentCodeFunctions.getInfoExperimentID(space, project);
247     String INFO_SAMPLE_CODE = project + "000";
248 
249     List<Sample> samples = openbis.getSamplesOfProject("/" + space + "/" + project);
250     Set<TechnologyType> techs = new HashSet<>();
251     int size = samples.size();
252     logger.info("converting " + project);
253     logger.info(size + " samples");
254     logger.info("target experiment: " + TARGET_EXPERIMENT);
255     Experiment exp = openbis.getExperimentById(TARGET_EXPERIMENT);
256 
257     boolean exists = exp != null;
258     boolean sampleExists = openbis.sampleExists(INFO_SAMPLE_CODE);
259 
260     logger.info("experiment exists: " + exists);
261     logger.info("sample exists: " + sampleExists);
262 
263     for (Sample s : samples) {
264       String type = s.getType().getCode();
265       if (type.equals("Q_TEST_SAMPLE")) {
266         String analyte = s.getProperties().get("Q_SAMPLE_TYPE");
267         TechnologyType tech = ParserHelpers.typeToTechnology.get(analyte);
268         if (tech != null)
269           techs.add(tech);
270       }
271       if (types.contains(type)) {
272         String code = s.getCode();
273         XMLParser par = new XMLParser();
274         List<Property> props = par.getAllPropertiesFromXML(s.getProperties().get("Q_PROPERTIES"));
275         for (Property p : props) {
276           String lab = p.getLabel();
277           String val = p.getValue();
278           String unit = null;
279           if (p.hasUnit())
280             unit = p.getUnit().getValue();
281           if (p.getType().equals(PropertyType.Factor)) {
282 
283             Pair<String, String> valunit = new ImmutablePair<String, String>(val, unit);
284             if (expDesign.containsKey(lab)) {
285               Map<Pair<String, String>, List<String>> levels = expDesign.get(lab);
286               if (levels.containsKey(valunit)) {
287                 levels.get(valunit).add(code);
288               } else {
289                 levels.put(valunit, new ArrayList<String>(Arrays.asList(code)));
290               }
291             } else {
292               Map<Pair<String, String>, List<String>> newLevel =
293                   new HashMap<Pair<String, String>, List<String>>();
294               newLevel.put(valunit, new ArrayList<String>(Arrays.asList(code)));
295               expDesign.put(lab, newLevel);
296             }
297 
298           } else {
299             Qproperty newProp = null;
300             if (p.hasUnit()) {
301               newProp = new Qproperty(code, p.getLabel(), p.getValue(), p.getUnit());
302             } else {
303               newProp = new Qproperty(code, p.getLabel(), p.getValue());
304             }
305             if (otherProps.containsKey(code)) {
306               otherProps.get(code).add(newProp);
307             } else {
308               otherProps.put(code, new ArrayList<Qproperty>(Arrays.asList(newProp)));
309             }
310           }
311         }
312       }
313     }
314     StudyXMLParser p = new StudyXMLParser();
315     List<TechnologyType> techTypes = new ArrayList<>(techs);
316     JAXBElement<Qexperiment> res =
317         p.createNewDesign(new HashSet<>(), techTypes, expDesign, otherProps);
318     String xml = p.toString(res);
319     Map<String, Object> props = new HashMap<>();
320     String infoCode = project + "_INFO";
321     props.put("Q_EXPERIMENTAL_SETUP", xml);
322     if (!exists) {
323       logger.info("creating new experiment");
324       creator.registerExperiment(space, project, ExperimentType.Q_PROJECT_DETAILS, infoCode, props);
325     } else {
326       logger.info("updating existing experiment");
327 
328       creator.updateExperiment(TARGET_EXPERIMENT, props);
329     }
330     if (!sampleExists) {
331       logger.info("registering info sample");
332       ISampleBean infoSample = new TSVSampleBean(INFO_SAMPLE_CODE, infoCode, project, space,
333           SampleType.Q_ATTACHMENT_SAMPLE, "", new ArrayList<String>(),
334           new HashMap<String, Object>());
335       creator.registerSampleBatch(new ArrayList<>(Arrays.asList(infoSample)));
336     }
337     logger.info(project + " done");
338   }
