45    public void createAttachmentFromStringMoveAndMarker(String content, String fileName,
46        String description, String type, String user, String barcode) {
47      Date date = new java.util.Date();
48      String timeStamp =
49          new Timestamp(date.getTime()).toString().split(" ")[1].replace(":", "").replace(".", "");
50  
51      AttachmentInformation attachment =
52          new AttachmentInformation(fileName, description, type, user, barcode, timeStamp);
53      Sardine sardine = SardineFactory.begin(config.getUser(), config.getPass());
54  
55      String folder = attachment.getFolder();
56      byte[] data = content.getBytes(StandardCharsets.UTF_8);
57      try {
58        sardine.createDirectory(config.getUri() + timeStamp);
59  
60        sardine.put(config.getUri() + folder + "/" + attachment.getFileName(), data);
61  
62        byte[] metadata = createMetadataString(attachment).getBytes();
63        sardine.put(config.getUri() + folder + "/" + "metadata.txt", metadata);
64  
65        Thread.sleep(1000);
66  
67        createMarker(sardine, folder, attachment.getFileName());
68  
69      } catch (Exception e1) {
70        e1.printStackTrace();
71        try {
72          sardine.shutdown();
73        } catch (IOException e) {
74          e.printStackTrace();
75        }
76      }
77    }

79    public void moveAttachment(Sardine sardine, AttachmentInformation attachment) {
80      byte[] data;
81      try {
82        sardine.createDirectory(config.getUri() + attachment.getFolder());
83  
84        data = FileUtils.readFileToByteArray(new File(tmpFolder + attachment.getName()));
85        sardine.put(config.getUri() + attachment.getFolder() + "/" + attachment.getFileName(), data);
86  
87        byte[] metadata = createMetadataString(attachment).getBytes();
88        sardine.put(config.getUri() + attachment.getFolder() + "/" + "metadata.txt", metadata);
89  
90      } catch (Exception e1) {
91        e1.printStackTrace();
92        try {
93          sardine.shutdown();
94        } catch (IOException e) {
95          e.printStackTrace();
96        }
97      }
98    }

100   /**
101    * Moves attachments to the datamover folder.
102    * 
103    * @param attachments List of names and other infos for each attachment
104    * @param moveUploadsReadyRunnable
105    * @param object2
106    * @param object
107    * @return
108    * @throws IOException
109    */
110   public void moveAttachments(final List<AttachmentInformation> attachments, final ProgressBar bar,
111       final Label info, final MoveUploadsReadyRunnable ready) throws IOException {
112     final Sardine sardine = SardineFactory.begin(config.getUser(), config.getPass());
113     ready.setSardine(sardine);
114     final int todo = attachments.size();
115     if (todo > 0) {
116       Thread t = new Thread(new Runnable() {
117         volatile int current = 0;
118 
119         @Override
120         public void run() {
121           for (AttachmentInformation a : attachments) {
122             current++;
123             double frac = current * 1.0 / todo;
124             UI.getCurrent().access(new UpdateProgressBar(bar, info, frac));
125             moveAttachment(sardine, a);
126           }
127           UI.getCurrent().access(ready);
128           UI.getCurrent().setPollInterval(-1);
129           try {
130             createMarkers(attachments);
131           } catch (InterruptedException e) {
132             e.printStackTrace();
133           } catch (IOException e) {
134             e.printStackTrace();
135           }
136         }
137       });
138       t.start();
139       UI.getCurrent().setPollInterval(100);
140     } else {
141       UI.getCurrent().access(ready);
142     }
143   }

145   protected String createMetadataString(AttachmentInformation a) {
146     String user = a.getUser();
147     String info = a.getInfo().replace("\n", " ");
148     String barcode = a.getBarcode();
149     String type = a.getType();
150     return "user=" + user + "\n" + "info=" + info + "\n" + "barcode=" + barcode + "\n" + "type="
151         + type;
152   }

154   private void createMarker(Sardine sardine, String folder, String fileName)
155       throws InterruptedException, IOException {
156     String prefix = ".MARKER_is_finished_";
157     String datafile = folder + "/" + fileName;
158     byte[] data = new byte[0];
159     int maxTries = 2;
160     while (maxTries > 0 && !sardine.exists(config.getUri() + datafile)) {
161       maxTries--;
162       Thread.sleep(100);
163     }
164     if (!sardine.exists(config.getUri() + datafile)) {
165       logger.error("Did not create marker file because folder " + datafile
166           + " was not found on the server.");
167       logger.error("Killing Sardine process.");
168       sardine.shutdown();
169     } else {
170       // data = FileUtils.readFileToByteArray(marker);
171       sardine.put(config.getUri() + prefix + folder, data);
172     }
173   }
174 
175   private void createMarkers(List<AttachmentInformation> attachments)
176       throws InterruptedException, IOException {
177     Sardine sardine = SardineFactory.begin(config.getUser(), config.getPass());
178     for (AttachmentInformation a : attachments) {
179       createMarker(sardine, a.getFolder(), a.getFileName());
180     }
181     sardine.shutdown();
182   }
