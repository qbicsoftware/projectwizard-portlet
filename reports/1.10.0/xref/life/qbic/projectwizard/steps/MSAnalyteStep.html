301   private void initTableListenersForWashSelection(Table t, SampleSelectComponent sampleSelect,
302       String infoCol, int washRow) {
303 
304     t.addValueChangeListener(new ValueChangeListener() {
305       @Override
306       public void valueChange(ValueChangeEvent event) {
307         boolean mainTable = infoCol.equals("Sample");
308         boolean valid = false;
309         Object id = t.getValue();
310         if (mainTable) {
311           valid = rowIsMeasuredAndNotNull(id);
312         } else {
313           if (t.equals(msEnrichmentTable.getTable()))
314             valid = msEnrichmentTable.rowIsMeasuredAndNotNull(id);
315           else
316             valid = msFractionationTable.rowIsMeasuredAndNotNull(id);
317         }
318         if (valid) {
319           String info = "";
320           if (infoCol.equals("Sample")) {
321             Label l = parseLabelRow(t, id, "Sample");
322             sampleSelect.setSample(l, t, id);
323             info = l.getValue();
324           } else {
325             TextField txt = parseTextRow(t, id, infoCol);
326             sampleSelect.setSample(txt, t, id);
327             info = txt.getValue();
328           }
329           String info2 = info;
330           if (info.contains("<br>")) {
331             String[] splt = info.split("<br>");
332             info = splt[0];
333             if (splt.length > 1)
334               info2 = splt[1];
335           }
336           parseTextRow(washRuns, washRow, "Name").setValue(info + " wash");
337           parseTextRow(washRuns, washRow, "Lab ID").setValue(info2 + " wash");
338         } else {
339           notifyUnmeasuredSample();
340         }
341         cancelListeners();
342         cancelTableSelections();
343       }
344     });
345     t.setSelectable(true);
346   }

368   protected boolean rowIsMeasuredAndNotNull(Object id) {
369     if (id == null)
370       return false;
371     ComboBox box = parseBoxRow(baseAnalyteSampleTable, id, "Process");
372     if (box.getValue() != null) {
373       String option = (String) box.getValue();
374       return option.equals("Both") || option.equals("Measure");
375     } else
376       return false;
377   }

379   protected void cancelListeners() {
380     List<Table> tables = Arrays.asList(baseAnalyteSampleTable, msFractionationTable.getTable(),
381         msEnrichmentTable.getTable());
382     for (Table t : tables) {
383       Collection<?> listeners = t.getListeners(ValueChangeEvent.class);
384       for (Object l : listeners) {
385         t.removeValueChangeListener((ValueChangeListener) l);
386       }
387     }
388   }

390   protected void cancelTableSelections() {
391     List<Table> tables = Arrays.asList(baseAnalyteSampleTable, msFractionationTable.getTable(),
392         msEnrichmentTable.getTable());

394       t.select(null);
395       t.setSelectable(false);
396     }

853     ExperimentModel baseAnalytes = new ExperimentModel(-5);

931                                                                                         // than one

933                                                                                         // needed?
934       OpenbisMSSample washRun =
935           new OpenbisMSSample((int) i, new ArrayList<AOpenbisSample>(Arrays.asList(parent)), name,
936               labID, new ArrayList<life.qbic.xml.properties.Property>(), "");
